01 엔진 아키텍처

Shader_Deferred.hlsl - Directional Light HLSL 
PS_OUT_LIGHT PS_MAIN_DIRECTIONAL(PS_IN In)
{
    PS_OUT_LIGHT Out = (PS_OUT_LIGHT) 0;

    vector vDepthDesc = g_DepthTexture.Sample(PointSampler, In.vTexcoord);
    vector vNormalDesc = g_NormalTexture.Sample(PointSampler, In.vTexcoord);
    vector vNormal = vector(vNormalDesc.xyz * 2.f - 1.f, 0.f);

    // Diffuse 라이팅
    Out.vShade = g_vLightDiffuse * saturate(max(dot(normalize(g_vLightDir) * -1.f,
        normalize(vNormal)), 0.f) + (g_vLightAmbient * g_vMtrlAmbient));

    // 월드 좌표 복원 (Depth → World Position)
    vector vWorldPos;
    vWorldPos.x = In.vTexcoord.x * 2.f - 1.f;
    vWorldPos.y = In.vTexcoord.y * -2.f + 1.f;
    vWorldPos.z = vDepthDesc.x;
    vWorldPos.w = 1.f;
    vWorldPos = vWorldPos * (vDepthDesc.y * 3000.f);
    vWorldPos = mul(vWorldPos, g_ProjMatrixInv);
    vWorldPos = mul(vWorldPos, g_ViewMatrixInv);

    // Specular 반사
    vector vReflect = reflect(normalize(g_vLightDir), normalize(vNormal));
    vector vLook = vWorldPos - g_vCamPosition;
    Out.vSpecular = (g_vLightSpecular * g_vMtrlSpecular) *
        pow(max(dot(normalize(vLook) * -1.f, normalize(vReflect)), 0.f), 30.f);

    return Out;
}
Shader_Deferred.hlsl - Point Light (거리 감쇠) HLSL 
PS_OUT_LIGHT PS_MAIN_POINT(PS_IN In)
{
    PS_OUT_LIGHT Out = (PS_OUT_LIGHT) 0;

    // G-Buffer에서 월드 좌표 복원
    vector vDepthDesc = g_DepthTexture.Sample(PointSampler, In.vTexcoord);
    vector vWorldPos;
    vWorldPos.x = In.vTexcoord.x * 2.f - 1.f;
    vWorldPos.y = In.vTexcoord.y * -2.f + 1.f;
    vWorldPos.z = vDepthDesc.x;
    vWorldPos.w = 1.f;
    vWorldPos = vWorldPos * (vDepthDesc.y * 3000.f);
    vWorldPos = mul(vWorldPos, g_ProjMatrixInv);
    vWorldPos = mul(vWorldPos, g_ViewMatrixInv);

    // 거리 기반 감쇠 (Attenuation)
    vector vLightDir = vWorldPos - g_vLightPos;
    float fDistance = length(vLightDir);
    float fAtt = saturate((g_fLightRange - fDistance) / g_fLightRange);

    Out.vShade = g_vLightDiffuse * saturate(max(dot(normalize(vLightDir) * -1.f,
        normalize(vNormal)), 0.f) + (g_vLightAmbient * g_vMtrlAmbient)) * fAtt;

    return Out;
}

03 포스트 프로세싱

모듈러 글리치 렌즈 아키텍처

"글리치 셰이더"를 하나의 거대한 블롭으로 만들지 않고, 파워 / 발생률 / 속도 / 색수차 강도 / 레드블록 혼합률을 각각 독립 파라미터로 분리해서 GameInstance 가 노출하도록 설계했습니다. 호출부에서 상황에 맞게 강도만 조절하면, 같은 셰이더가 피격 / 스킬 / 컷씬 / 죽음 직전 등 다른 표현을 만들어 냅니다.

글리치 셰이더 본체

위 파라미터들이 실제로 들어가는 셰이더입니다. 블록 노이즈 + 색수차 + 레드 글리치 36개 오버레이를 하나의 픽셀 셰이더에서 합성합니다.

Shader_Deferred.hlsl - 글리치 셰이더 HLSL 
PS_OUT PS_MAIN_GLITCH(PS_IN In)
{
    PS_OUT Out = (PS_OUT) 0;

    // 블록 단위 랜덤 흔들림
    float enable_shift = (random(float2(floor(TIME * shake_speed), 0.0))
        < shake_rate) ? 1.0f : 0.0f;

    float2 fixed_uv = In.vTexcoord;
    fixed_uv.x += (random(float2(floor(In.vTexcoord.y * shake_block_size)
        / shake_block_size, TIME)) - 0.5f) * shake_power * enable_shift;

    float4 pixel_color = g_DeferredTexture.SampleLevel(LinearSampler, fixed_uv, 0.0f);

    // 색수차 (Chromatic Aberration) - R/B 채널 분리
    pixel_color.r = lerp(pixel_color.r, g_DeferredTexture.SampleLevel(LinearSampler,
        fixed_uv + float2(shake_color_rate, 0.0f), 0.0f).r, enable_shift);
    pixel_color.b = lerp(pixel_color.b, g_DeferredTexture.SampleLevel(LinearSampler,
        fixed_uv + float2(-shake_color_rate, 0.0f), 0.0f).b, enable_shift);

    // 레드 글리치 블록 오버레이
    if (g_bUseRedGlitch == true) {
        for (int i = 0; i < 36; ++i) {
            float2 center = float2(random(float2(TIME, i * 0.1f)) * g_ScreenSize.x,
                i * red_block_height * 1.5f * g_ScreenSize.y / 720.f);
            float rect_width = red_block_min_length + (red_block_max_length
                - red_block_min_length) * random(float2(TIME, i * 0.3f));
            // ... 사각형 영역 내 빨간색 혼합
            pixel_color.rgb = lerp(pixel_color.rgb,
                float3(1.0f, 0.0f, 0.0f), red_mix_rate);
        }
    }

    Out.vColor = pixel_color;
    return Out;
}

방사형 블러 (Radial Blur)

대시, 와이어 액션 등 고속 이동 시 화면 중심에서 바깥으로 퍼지는 방사형 블러를 적용합니다. 샘플 수와 블러 강도를 실시간으로 조절할 수 있습니다.

Shader_Deferred.hlsl - 방사형 블러 HLSL 
PS_OUT PS_MAIN_BLUR(PS_IN In)
{
    PS_OUT Out = (PS_OUT) 0;
    float4 pixel_color = g_DeferredTexture.Sample(LinearSampler, In.vTexcoord);

    float2 direction = In.vTexcoord - g_Center;
    float distance = length(direction);

    int adjustedNumSamples = max(8, (int)(numSamples * g_BlurStrength));
    float totalWeight = 0.0f;
    float4 blurredColor = float4(0, 0, 0, 0);

    for (int i = 0; i < adjustedNumSamples; ++i) {
        float sampleOffset = (float(i) / float(adjustedNumSamples)) * distance;
        float2 sampleCoord = In.vTexcoord + normalize(direction) * sampleOffset;
        float weight = 1.0f - (sampleOffset / distance);
        blurredColor += g_DeferredTexture.Sample(LinearSampler, sampleCoord) * weight;
        totalWeight += weight;
    }

    blurredColor /= totalWeight;
    Out.vColor = lerp(pixel_color, blurredColor, distance);
    return Out;
}

Glow / Bloom

가우시안 가중치 기반의 2-Pass 블러(수평 + 수직)로 발광 오브젝트에 Bloom 효과를 적용합니다. 무기 트레일, 그래플링 포인트, 적 공격 이펙트 등에 사용됩니다.

Shader_Deferred.hlsl - Glow 가우시안 블러 HLSL 
PS_OUT PS_MAIN_GLOW(PS_IN In)
{
    PS_OUT Out = (PS_OUT) 0;

    float2 texelSize = 1.0 / g_ScreenSize * 2.f;
    float4 Horizon_color = g_GlowTexture.Sample(LinearSampler, In.vTexcoord) * weights[0];

    for (int i = 1; i < 7; ++i) {
        Horizon_color += g_GlowTexture.Sample(LinearSampler,
            In.vTexcoord + float2(i * texelSize.x, 0.0)) * weights[i];
        Horizon_color += g_GlowTexture.Sample(LinearSampler,
            In.vTexcoord - float2(i * texelSize.x, 0.0)) * weights[i];
    }

    Out.vColor = Horizon_color;
    return Out;
}

Katana.cpp - 본 부착 & 트레일 활성화 C++ 
void CKatana::Tick(_float fTimeDelta)
{
    m_pTrailCom->Set_TrailOn(false);

    // 공격 중일 때만 트레일 활성화
    if ((PLAYER->Get_IsAttack() == true || m_bOn))
        m_pTrailCom->Set_TrailOn(true);

    // 본 매트릭스에서 무기 위치 동기화
    _matrix BoneMatrix = XMLoadFloat4x4(m_pBoneMatrix);
    for (size_t i = 0; i < 3; i++)
        BoneMatrix.r[i] = XMVector3Normalize(BoneMatrix.r[i]);

    SetUp_WorldMatrix(m_pTransformCom->Get_WorldMatrix() * BoneMatrix);
}

적 사격 / 슬래시 / 패리 순간에 스폰되는 임팩트 이펙트입니다. 모두 컨테이너 풀(CONTAIN_ACTIVE / SLEEP)에서 꺼내 쓰고, 라이프 타임이 끝나면 풀로 돌아갑니다. 사망 직전 보스가 마구잡이로 휘두를 때처럼 이펙트가 폭증하는 구간에서도 할당 지연이 없습니다.

패리 윈도우 — 특정 페이즈에서만 받아치기 가능

고스트 러너의 핵심 재미는 "보스 공격을 정확한 타이밍에 받아쳐서 카운터를 넣는" 패리 전투입니다. 단순히 "공격 중이면 패리 가능" 으로 만들면 너무 쉬워서, 보스 패턴의 특정 페이즈(m_iPatternPhase == 2)에서만 패리가 성립하도록 충돌 콜백에서 4개 조건을 동시에 검사합니다.

PlayerManager.cpp - 와이어 액션 구현 C++ 
void CPlayer_Manager::State_WireAction(_float fTimeDelta)
{
    switch(eWireState) {
    case WIRE_START: {
        m_pRigidBody->Set_Grab(CRigidbody::WIRE_START);
        m_fWireTime = m_fMaxWireTime;

        // 타겟 방향으로 3축 속도 계산
        m_pRigidBody->Set_VelocityZ(
            XMVectorGetZ(m_vGrabTargetDir) * fabs(m_fGrabZDistance) * 0.04f);
        m_pRigidBody->Set_VelocityX(
            XMVectorGetX(m_vGrabTargetDir) * fabs(m_fGrabXDistance) * 0.04f);
        m_pRigidBody->Set_VelocityY(
            XMVectorGetY(m_vGrabTargetDir) * fabs(m_fGrabYDistance) * 0.1f);
        m_pRigidBody->Set_Ground(false);

        eWireState = WIRE_LOOP;
        break;
    }
    case WIRE_LOOP: {
        m_fWireTime -= fTimeDelta;
        // 타겟 근처 도달 또는 시간 초과 시 ATTACH로 전환
        if (m_fWireTime <= 0.f ||
            fabs(Get_PlayerToTargetXZDistance(...)) < 8.f) {
            eWireState = WIRE_ATTACH;
        }
        break;
    }
    }
}

m_fTimer += fTimeDelta;
if (m_fTimer >= 0.016f)
{
    m_fTimer -= 0.016f;                 // 60Hz 고정 샘플링
    ++m_iStartingPoint;

    _float3 vUpperPos = { resultUpper._41, resultUpper._42, resultUpper._43 };
    _float3 vDownPos  = { resultDown._41,  resultDown._42,  resultDown._43  };

    if (m_iStartingPoint >= m_iMaxRect)
    {
        m_iStartingPoint = m_iMaxRect;
        m_vecUpperPositions.pop_front();   // 오래된 점은 잘라내고
        m_vecDownPositions.pop_front();
        m_vecUpperPositions.push_back(vUpperPos);  // 새 점은 뒤에 추가
        m_vecDownPositions.push_back(vDownPos);
    }
    else
    {
        m_vecUpperPositions[m_iStartingPoint] = vUpperPos;
        m_vecDownPositions[m_iStartingPoint] = vDownPos;
    }
    Update_Buffer();
}
Update_Buffer — D3D11_MAP_WRITE_NO_OVERWRITE 로 스톨 회피 C++ 
void CVIBuffer_TrailRect::Update_Buffer()
{
    D3D11_MAPPED_SUBRESOURCE SubResource{};
    m_pContext->Map(m_pVB, 0, D3D11_MAP_WRITE_NO_OVERWRITE, 0, &SubResource);
    // ... 정점 데이터 갱신
    m_pContext->Unmap(m_pVB, 0);
}

핵심 디테일 — NO_OVERWRITE 의 의미: D3D11_MAP_WRITE_DISCARD 는 GPU 가 쓰던 버퍼를 통째로 버리고 새 메모리를 할당하지만, WRITE_NO_OVERWRITE 는 "GPU 가 지금 안 읽는 영역에만 덮어쓴다"고 약속하는 모드입니다. 매 프레임 정점 일부만 갱신하는 트레일에 가장 적합하고, 드라이버 동기화 비용이 거의 없습니다.

본 매트릭스 비균등 스케일 보정

스켈레탈 모델의 본 매트릭스는 부모 본에서 비균등 스케일(non-uniform scale)이 누적될 수 있습니다. 그 상태로 무기를 본에 부착하면 트레일이 늘어지거나 찌그러지는 현상이 나타납니다.

이를 막기 위해 무기 매트릭스의 회전 행 3개를 매 프레임 정규화해서, 스케일이 1.0 으로 강제된 회전만 추출합니다. 단순한 처리 같지만 트레일 품질을 결정적으로 좌우합니다.

Katana.cpp — 본 매트릭스 회전 행 정규화 C++ 
void CKatana::Tick(_float fTimeDelta)
{
    m_pTrailCom->Set_TrailOn(PLAYER->Get_IsAttack() || m_bOn);

    SetUp_WorldMatrix(m_pTransformCom->Get_WorldMatrix() * BoneMatrix);
}

프레임 변동에 따라 트레일이 끊기거나 듬성듬성해지는 문제
매 프레임 정점을 추가하던 방식을 16ms 고정 샘플링 + deque 링 버퍼로 바꿨습니다. 동시에 동적 정점 버퍼 갱신에 D3D11_MAP_WRITE_NO_OVERWRITE 플래그를 사용해 드라이버 동기화 비용을 제거했습니다.
프레임이 30 이든 144 든 트레일이 항상 일정한 매끄러움을 유지하고, 매 프레임 GPU 스톨 없이 정점을 갱신할 수 있게 되었습니다.
본에 부착한 무기 트레일이 늘어지거나 찌그러지는 현상
부모 본에서 누적된 비균등 스케일이 무기 매트릭스에 섞여 들어오는 것이 원인이었습니다. 본 매트릭스의 회전 행 3개를 매 프레임 XMVector3Normalize 로 정규화해서 스케일 성분을 제거하는 처리를 추가했습니다.
한 줄짜리 정규화로 트레일이 항상 일정한 폭으로 깨끗하게 그려지게 되었습니다. 외부에서 보이지 않는 디테일이지만 영상 퀄리티의 마지막 5% 를 결정합니다.
"패리가 너무 쉽거나 너무 어렵다" 는 피드백
"공격 중 + 충돌" 만으로 패리를 허용하던 초기 구현을 폐기하고, 보스의 PatternState + PatternType + PatternPhase 까지 4중 검사로 좁혔습니다. 페이즈2 라는 짧은 윈도우에서만 받아치기가 성립하도록 만들어, 보스의 모션을 읽어야 패리가 가능해졌습니다.
전투 템포가 "들이대기" 에서 "관찰 후 받아치기" 로 바뀌어, 고스트 러너 특유의 리듬감이 살아났습니다.
화면 흔들림과 글리치 이펙트가 어디서나 똑같이 보여서 단조로움
글리치 셰이더의 모든 파라미터(파워 / 발생률 / 색수차 / 레드블록 혼합률 / 셰이크 강도 / 지속시간)를 GameInstance 인터페이스로 분리해서, 호출부가 상황별로 자유롭게 조합할 수 있게 만들었습니다.
셰이더는 하나만 유지하면서 피격 / 스킬 / 저체력 / 죽음 직전 등이 모두 다른 시각 표현을 갖게 되었고, 디자이너가 코드를 건드리지 않고 톤 조절이 가능해졌습니다.
이펙트가 폭증하는 구간에서 발생하는 할당 지연
Bullet / Beam / ShockWave / Spark 같은 빈번하게 생성/소멸되는 오브젝트들을 컨테이너 풀로 묶고, CONTAIN_ACTIVE / CONTAIN_SLEEP 상태로만 토글하도록 변경했습니다. 사망 직전 보스의 마구잡이 공격 같은 폭증 구간에서도 새 할당이 발생하지 않습니다.
전투 클라이맥스에서도 프레임이 일정하게 유지되고, 메모리 단편화로 인한 후반부 성능 저하가 사라졌습니다.
