Part 1: 스케일링의 필요성

왜 서버 스케일링이 중요한가요?

게임 출시, 이벤트 등으로 트래픽이 급증할 때 서버가 버텨야 합니다.

게임 트래픽 패턴 (동시접속자 vs 시간)
출시 피크
예상의 10배 접속
이벤트
갑자기 2배 접속
안정화
평상시 트래픽
해결: 동적 스케일링
필요할 때 늘리고, 줄이고
출시 피크 → 서버 다운 방지
평상시 → 비용 낭비 방지

게임 트래픽은 출시, 이벤트에 따라 급변하며 동적 스케일링이 필수

수직 vs 수평 확장이 뭔가요?
특성 수직 확장 (Scale Up) 수평 확장 (Scale Out)
방법 더 좋은 서버로 교체 서버 대수를 늘림
예시 4코어 → 32코어 서버 1대 → 10대
장점 간단, 코드 변경 불필요 무한 확장 가능
단점 한계 있음, 비용 급증 분산 시스템 복잡도
적합 초기, 소규모 대규모, 장기 운영

대부분의 대규모 게임은 수평 확장을 기본으로 설계합니다.

class LoginServer {
    // 서버에 상태 없음
    // 모든 데이터는 DB/Redis에서 조회

public:
    LoginResponse HandleLogin(const LoginRequest& req) {
        // 1. 인증 확인 (DB 조회)
        auto user = db.GetUser(req.username);
        if (!user || !VerifyPassword(req.password, user.hash)) {
            return { false, "Invalid credentials" };
        }

        // 2. 세션 생성 (Redis에 저장)
        string sessionId = GenerateSessionId();
        redis.Set("session:" + sessionId, user.id, TTL_1_HOUR);

        // 3. 응답 (서버에 아무것도 저장 안 함)
        return { true, sessionId };
    }
};

// - 서버 추가/제거가 자유로움
// - 한 서버 죽어도 다른 서버로 요청 가능

// Stateful: 서버에 게임 상태 유지
// 특정 매치는 특정 서버에서만 처리

class GameServer {
    // 서버 메모리에 게임 상태 유지
    std::map activeMatches;

public:
    void HandlePlayerAction(PlayerID player, const Action& action) {
        // 1. 이 플레이어가 속한 매치 찾기
        auto match = FindMatchForPlayer(player);
        if (!match) return;

        // 2. 게임 상태 업데이트 (메모리에서)
        match->ApplyAction(player, action);

        // 3. 변경사항 브로드캐스트
        BroadcastToMatch(match->id, match->GetSnapshot());
    }
};

// 단점:
// - 플레이어는 항상 같은 서버로 가야 함 (Sticky Session)
// - 서버 죽으면 진행 중인 게임 손실

Graceful Shutdown 구현
class GameServerLifecycle {
    enum class State {
        Running,      // 정상 운영
        Draining,     // 신규 매치 안 받음
        Terminating   // 종료 중
    };

    State state = State::Running;

public:
    // Scale-down 시 호출됨
    void OnShutdownSignal() {
        state = State::Draining;
        Log("Server draining - no new matches accepted");

        // 오케스트레이터에게 알림 (새 매치 안 보내도록)
        NotifyOrchestrator(ServerStatus::Draining);

        // 진행 중인 매치가 끝날 때까지 대기
        WaitForMatchesToEnd();
    }

    void WaitForMatchesToEnd() {
        while (!activeMatches.empty()) {
            // 매치별 최대 시간 체크 (너무 오래 걸리면 강제 종료)
            for (auto& [id, match] : activeMatches) {
                if (match.duration > MAX_MATCH_DURATION) {
                    match.ForceEnd("Server maintenance");
                }
            }

            Sleep(1000);  // 1초마다 체크
        }

        // 모든 매치 종료됨
        state = State::Terminating;
        Log("All matches ended, shutting down");
        exit(0);
    }

    bool CanAcceptNewMatch() {
        return state == State::Running;
    }
};

// 스케일다운 흐름:
// 1. K8s가 SIGTERM 전송
// 2. 서버가 Draining 상태로 전환
// 3. 기존 매치만 처리
// 4. 모든 매치 종료 후 서버 종료

Part 4: 서버 풀 관리

매치를 어떤 서버에 배정하나요?
서버 오케스트레이터
class ServerOrchestrator {
    struct ServerInfo {
        string serverId;
        string address;
        int activeMatches;
        int maxMatches;
        float cpuUsage;
        float memUsage;
        Region region;
        ServerStatus status;
    };

    std::vector serverPool;

public:
    // 매치에 적합한 서버 선택
    ServerInfo* AllocateServer(const MatchRequest& request) {
        std::vector candidates;

        // 1. 조건에 맞는 서버 필터링
        for (auto& server : serverPool) {
            if (!IsServerEligible(server, request)) continue;
            candidates.push_back(&server);
        }

        if (candidates.empty()) {
            // 서버 부족 → 스케일업 트리거
            RequestScaleUp(request.region);
            return nullptr;
        }

        // 2. 최적 서버 선택 (bin packing)
        return SelectBestServer(candidates, request);
    }

private:
    bool IsServerEligible(const ServerInfo& server,
                         const MatchRequest& request) {
        // 지역 일치
        if (server.region != request.region) return false;

        // 상태 확인
        if (server.status != ServerStatus::Running) return false;

        // 용량 확인
        if (server.activeMatches >= server.maxMatches) return false;

        // 리소스 확인
        if (server.cpuUsage > 80 || server.memUsage > 80) return false;

        return true;
    }

    ServerInfo* SelectBestServer(std::vector& candidates,
                                 const MatchRequest& request) {
        // Bin Packing: 서버를 최대한 채워서 사용
        // (비어있는 서버를 줄여서 스케일다운 가능하게)

        std::sort(candidates.begin(), candidates.end(),
            [](ServerInfo* a, ServerInfo* b) {
                // 사용량 높은 순 (더 채워진 서버 우선)
                return a->activeMatches > b->activeMatches;
            });

        return candidates[0];
    }
};

class RegionRouter {
    struct DataCenter {
        Region region;
        string address;
        std::vector servers;
    };

    std::vector dataCenters = {
        { Region::KR, "kr.game.com", {} },  // 한국
        { Region::JP, "jp.game.com", {} },  // 일본
        { Region::US_WEST, "usw.game.com", {} },
        { Region::US_EAST, "use.game.com", {} },
        { Region::EU, "eu.game.com", {} },
    };

public:
    DataCenter* GetBestRegion(const std::vector& players) {
        // 방법 1: 지리적 위치 기반
        // IP → GeoIP로 위치 추정 → 가장 가까운 DC

        // 방법 2: 핑 측정 기반 (더 정확)
        std::map avgPings;

        for (const auto& dc : dataCenters) {
            int totalPing = 0;
            for (const auto& player : players) {
                totalPing += player.pingToRegion[dc.region];
            }
            avgPings[dc.region] = totalPing / players.size();
        }

        // 평균 핑이 가장 낮은 지역 선택
        Region bestRegion = std::min_element(
            avgPings.begin(), avgPings.end(),
            [](const auto& a, const auto& b) {
                return a.second < b.second;
            }
        )->first;

        return GetDataCenter(bestRegion);
    }

    // 클라이언트 측: 정기적으로 각 지역 핑 측정
    void MeasurePings() {
        for (const auto& dc : dataCenters) {
            auto start = Clock::now();
            Ping(dc.address);
            auto rtt = Clock::now() - start;
            myPingToRegion[dc.region] = rtt.count();
        }
    }
};

Consistent Hashing 샤딩
class ConsistentHashSharding {
    // 가상 노드 (한 샤드당 100개의 가상 노드)
    std::map ring;  // hash → shard_id
    int virtualNodesPerShard = 100;

public:
    void AddShard(int shardId) {
        for (int i = 0; i < virtualNodesPerShard; i++) {
            string key = "shard_" + std::to_string(shardId) +
                        "_vnode_" + std::to_string(i);
            uint64_t hash = MurmurHash(key);
            ring[hash] = shardId;
        }
    }

    void RemoveShard(int shardId) {
        auto it = ring.begin();
        while (it != ring.end()) {
            if (it->second == shardId) {
                it = ring.erase(it);
            } else {
                ++it;
            }
        }
    }

    int GetShard(uint64_t userId) {
        if (ring.empty()) return -1;

        uint64_t hash = MurmurHash(std::to_string(userId));

        // 해시값보다 크거나 같은 첫 번째 노드 찾기
        auto it = ring.lower_bound(hash);
        if (it == ring.end()) {
            it = ring.begin();  // 링 순환
        }

        return it->second;
    }
};

// Consistent Hashing 장점:
// - 샤드 추가/제거 시 최소한의 데이터만 이동
// - 균등 분배
// - 확장성 좋음

Part 6: 장애 대응

서버가 죽으면 어떻게 대응하나요?
서버 Health Check와 장애 복구
class ServerHealthMonitor {
    struct HealthCheck {
        bool isHealthy;
        int consecutiveFailures;
        float responseTime;
        time_t lastCheck;
    };

    std::map serverHealth;
    const int MAX_FAILURES = 3;
    const float HEALTH_CHECK_INTERVAL = 5.0f;  // 5초

public:
    void RunHealthChecks() {
        for (auto& [serverId, health] : serverHealth) {
            auto start = Clock::now();
            bool success = PingServer(serverId);
            float responseTime = (Clock::now() - start).count();

            if (success && responseTime < TIMEOUT) {
                health.isHealthy = true;
                health.consecutiveFailures = 0;
                health.responseTime = responseTime;
            } else {
                health.consecutiveFailures++;
                if (health.consecutiveFailures >= MAX_FAILURES) {
                    health.isHealthy = false;
                    HandleServerFailure(serverId);
                }
            }
            health.lastCheck = time(nullptr);
        }
    }

    void HandleServerFailure(const string& serverId) {
        Log("Server %s marked as unhealthy", serverId.c_str());

        // 1. 로드 밸런서에서 제거
        loadBalancer.RemoveServer(serverId);

        // 2. 진행 중인 매치 복구 시도
        auto affectedMatches = GetMatchesOnServer(serverId);
        for (auto& matchId : affectedMatches) {
            TryRecoverMatch(matchId);
        }

        // 3. 새 서버 요청 (스케일업)
        RequestReplacementServer(serverId);

        // 4. 알림 전송
        SendAlert("Server failure: " + serverId);
    }

    void TryRecoverMatch(MatchID matchId) {
        // 주기적으로 저장된 체크포인트에서 복구
        auto checkpoint = redis.Get("match_checkpoint:" + matchId);
        if (checkpoint) {
            // 새 서버에서 매치 재시작
            auto newServer = AllocateNewServer();
            newServer->RestoreMatch(matchId, checkpoint);

            // 플레이어들에게 재연결 안내
            NotifyPlayersToReconnect(matchId, newServer->address);
        } else {
            // 복구 불가 → 매치 취소
            CancelMatch(matchId, "Server error - match canceled");
        }
    }
};

class MatchCheckpoint {
    const float CHECKPOINT_INTERVAL = 30.0f;  // 30초마다

public:
    void PeriodicCheckpoint(const GameState& state) {
        // 직렬화
        auto data = Serialize(state);

        // 압축
        auto compressed = Compress(data);

        // Redis에 저장 (TTL 설정)
        string key = "match_checkpoint:" + state.matchId;
        redis.SetEx(key, compressed, CHECKPOINT_TTL);

        Log("Checkpoint saved for match %s (%d bytes)",
            state.matchId.c_str(), compressed.size());
    }

    std::optional LoadCheckpoint(MatchID matchId) {
        string key = "match_checkpoint:" + matchId;
        auto compressed = redis.Get(key);

        if (!compressed) return std::nullopt;

        auto data = Decompress(*compressed);
        return Deserialize(data);
    }
};

// 체크포인트 주의사항:
// - 자주 저장: 복구 시 손실 적음, 하지만 부하 증가
// - 드물게 저장: 부하 적음, 하지만 복구 시 손실 큼
// - 중요 이벤트 시 추가 저장 (킬, 점령 등)

📌 핵심 정리

  • 수평 확장: 서버 대수를 늘려 용량 증가 (선호됨)
  • 수직 확장: 더 좋은 서버로 교체 (한계 있음)
  • Stateless: 상태 없는 서버, 쉬운 확장 (로그인, 로비)
  • Stateful: 상태 있는 서버, 복잡한 확장 (게임 서버)
  • Auto Scaling: 부하에 따라 자동 서버 추가/제거
  • Graceful Shutdown: 진행 중인 매치 보호하며 종료
  • 데이터 샤딩: Consistent Hashing으로 DB 분산
  • 장애 복구: Health Check, 체크포인트로 빠른 복구
