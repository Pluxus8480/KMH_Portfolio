Part 2: Future와 Promise

C++에서 비동기 결과를 어떻게 받나요?
std::future와 std::async
#include 

// 가장 간단한 비동기 실행
std::future loadTextureAsync(const char* path) {
    return std::async(std::launch::async, [path] {
        return loadTextureFromDisk(path);  // 백그라운드 실행
    });
}

void gameLoop() {
    // 로딩 시작
    auto textureFuture = loadTextureAsync("player.png");

    while (gameRunning) {
        // 게임 업데이트/렌더 (블로킹 없음)
        update();
        render();

        // 로딩 완료 확인 (논블로킹)
        if (textureFuture.valid() &&
            textureFuture.wait_for(std::chrono::milliseconds(0)) ==
            std::future_status::ready) {

            Texture* tex = textureFuture.get();  // 결과 획득
            applyTexture(tex);
        }
    }
}

// Promise로 직접 제어
std::promise promise;
std::future future = promise.get_future();

std::thread worker([&promise] {
    int result = heavyComputation();
    promise.set_value(result);  // 결과 설정
});

// 다른 곳에서
int value = future.get();  // 결과 대기

Part 3: 콜백 패턴

콜백 방식은 어떻게 구현하나요?
콜백 기반 비동기
// 콜백 타입 정의
using LoadCallback = std::function;

class AsyncLoader {
    struct LoadRequest {
        std::string path;
        LoadCallback callback;
    };

    std::queue pendingRequests_;
    std::queue> completedResults_;
    std::thread loaderThread_;
    std::mutex mutex_;
    std::condition_variable cv_;
    bool running_ = true;

public:
    // 비동기 로딩 요청
    void loadAsync(const char* path, LoadCallback callback) {
        std::lock_guard lock(mutex_);
        pendingRequests_.push({path, callback});
        cv_.notify_one();
    }

    // 메인 스레드에서 호출: 완료된 콜백 실행
    void processCallbacks() {
        std::queue> toProcess;
        {
            std::lock_guard lock(mutex_);
            std::swap(toProcess, completedResults_);
        }

        while (!toProcess.empty()) {
            auto& [tex, callback] = toProcess.front();
            callback(tex);  // 메인 스레드에서 안전하게 실행
            toProcess.pop();
        }
    }

private:
    void loaderLoop() {
        while (running_) {
            LoadRequest request;
            {
                std::unique_lock lock(mutex_);
                cv_.wait(lock, [this] {
                    return !pendingRequests_.empty() || !running_;
                });
                if (!running_) break;
                request = pendingRequests_.front();
                pendingRequests_.pop();
            }

            // 백그라운드에서 로딩
            Texture* tex = loadTextureFromDisk(request.path.c_str());

            // 결과 큐에 추가
            {
                std::lock_guard lock(mutex_);
                completedResults_.push({tex, request.callback});
            }
        }
    }
};

// 사용 예시
AsyncLoader loader;

loader.loadAsync("enemy.png", [](Texture* tex) {
    // 메인 스레드에서 실행됨
    enemy->setTexture(tex);
});

// 게임 루프에서
void gameLoop() {
    while (running) {
        loader.processCallbacks();  // 완료된 로딩 처리
        update();
        render();
    }
}

Part 4: 코루틴 (C++20)

코루틴은 비동기를 어떻게 단순화하나요?
C++20 Coroutine
#include 

// 간단한 Task 코루틴
template
struct Task {
    struct promise_type {
        T value;
        std::exception_ptr exception;

        Task get_return_object() {
            return Task{std::coroutine_handle::from_promise(*this)};
        }
        std::suspend_never initial_suspend() { return {}; }
        std::suspend_always final_suspend() noexcept { return {}; }
        void return_value(T v) { value = v; }
        void unhandled_exception() { exception = std::current_exception(); }
    };

    std::coroutine_handle handle;

    T get() {
        if (handle.promise().exception)
            std::rethrow_exception(handle.promise().exception);
        return handle.promise().value;
    }
};

// 비동기 대기 객체
struct AsyncLoad {
    const char* path;
    Texture* result = nullptr;

    bool await_ready() { return false; }  // 항상 중단

    void await_suspend(std::coroutine_handle<> h) {
        // 백그라운드에서 로딩 후 재개
        std::thread([this, h] {
            result = loadTextureFromDisk(path);
            h.resume();  // 코루틴 재개
        }).detach();
    }

    Texture* await_resume() { return result; }
};

// 코루틴 사용 예시 (동기 코드처럼 읽힘!)
Task loadGameAssets() {
    // 순차적으로 보이지만 비동기로 실행됨
    Texture* player = co_await AsyncLoad{"player.png"};
    Texture* enemy = co_await AsyncLoad{"enemy.png"};
    Texture* background = co_await AsyncLoad{"bg.png"};

    // 모든 로딩 완료 후 실행
    setupGame(player, enemy, background);
}

Part 5: 게임 에셋 스트리밍

오픈월드 게임의 스트리밍은 어떻게 하나요?
우선순위 기반 스트리밍
class StreamingManager {
    struct StreamRequest {
        std::string path;
        float priority;  // 거리 기반 우선순위
        std::function callback;

        bool operator<(const StreamRequest& o) const {
            return priority < o.priority;  // 높은 우선순위 먼저
        }
    };

    std::priority_queue requestQueue_;
    std::unordered_map loadedAssets_;
    std::unordered_set loadingAssets_;

public:
    void requestAsset(const char* path, float distance,
                     std::function callback) {
        // 이미 로드됨
        if (auto it = loadedAssets_.find(path); it != loadedAssets_.end()) {
            callback(it->second);
            return;
        }

        // 이미 로딩 중
        if (loadingAssets_.count(path)) return;

        // 우선순위 = 1 / (거리 + 1)
        float priority = 1.0f / (distance + 1.0f);
        requestQueue_.push({path, priority, callback});
    }

    void update() {
        // 프레임당 제한된 수의 로딩 시작
        const int MAX_LOADS_PER_FRAME = 2;
        int loadsStarted = 0;

        while (!requestQueue_.empty() && loadsStarted < MAX_LOADS_PER_FRAME) {
            auto request = requestQueue_.top();
            requestQueue_.pop();

            if (loadingAssets_.count(request.path)) continue;

            loadingAssets_.insert(request.path);
            startAsyncLoad(request);
            loadsStarted++;
        }
    }

private:
    void startAsyncLoad(const StreamRequest& request) {
        std::async(std::launch::async, [this, request] {
            Asset* asset = loadAssetFromDisk(request.path.c_str());

            // 메인 스레드에서 콜백
            postToMainThread([this, request, asset] {
                loadedAssets_[request.path] = asset;
                loadingAssets_.erase(request.path);
                request.callback(asset);
            });
        });
    }
};

📌 핵심 정리

  • 비동기 필요성: I/O가 게임 루프를 블로킹하면 프레임 드랍
  • std::future: 간단한 비동기 결과 획득, wait_for로 논블로킹 체크
  • 콜백 패턴: 결과를 메인 스레드에서 안전하게 처리
  • 코루틴: 비동기 코드를 동기처럼 작성 (C++20)
  • 스트리밍: 우선순위 큐 + LOD로 필요한 것만 점진적 로드
