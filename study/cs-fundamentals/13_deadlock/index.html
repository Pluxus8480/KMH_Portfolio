데드락의 원인과 해결법

Part 1: 데드락이란?

데드락은 어떻게 발생하나요?

두 개 이상의 스레드가 서로의 자원을 기다리며 영원히 대기하는 상태입니다.

데드락 발생 시나리오

순서Thread AThread B
1lock(mutexA) ✓
2lock(mutexB) ✓
3lock(mutexB) ... 대기 →
4lock(mutexA) ... 대기 →

결과: 양쪽 모두 영원히 대기!

Thread A

mutexA 보유

→ 대기 →
mutexB

Thread B가 보유

Thread B

mutexB 보유

→ 대기 →
mutexA

Thread A가 보유

순환 대기 (Circular Wait) 발생!

데드락 발생 조건은 무엇인가요?

4가지 조건이 모두 만족해야 데드락이 발생합니다.

조건 설명 게임 예시
상호 배제 자원을 한 번에 하나만 사용 Mutex로 보호된 자원
점유 대기 자원 보유하며 다른 자원 대기 Lock A 보유, Lock B 대기
비선점 강제로 자원 뺏기 불가 다른 스레드의 lock 해제 불가
순환 대기 자원 대기가 원형으로 연결 A→B→C→A 대기 체인

하나의 조건만 깨면 데드락을 방지할 수 있습니다!

Part 2: 데드락 방지 기법

순환 대기를 방지하는 방법은?
방법 1: Lock 순서 고정
// ❌ 데드락 가능
void transferBad(Account& from, Account& to, int amount) {
    std::lock_guard lock1(from.mutex);
    std::lock_guard lock2(to.mutex);  // 반대 순서면 데드락!
    // 이체 로직
}

// ✅ 항상 같은 순서로 lock (ID 기준)
void transferGood(Account& from, Account& to, int amount) {
    // 더 작은 ID를 먼저 lock
    Account* first = (from.id < to.id) ? &from : &to;
    Account* second = (from.id < to.id) ? &to : &from;

    std::lock_guard lock1(first->mutex);
    std::lock_guard lock2(second->mutex);
    // 이체 로직
}

// ✅ std::lock 사용 (원자적으로 여러 lock 획득)
void transferBest(Account& from, Account& to, int amount) {
    std::scoped_lock lock(from.mutex, to.mutex);  // C++17
    // 또는
    // std::lock(from.mutex, to.mutex);
    // std::lock_guard lg1(from.mutex, std::adopt_lock);
    // std::lock_guard lg2(to.mutex, std::adopt_lock);
}
방법 2: try_lock으로 점유 대기 방지
bool tryTransfer(Account& from, Account& to, int amount) {
    // 타임아웃 있는 시도
    std::unique_lock lock1(from.mutex, std::defer_lock);
    std::unique_lock lock2(to.mutex, std::defer_lock);

    // 둘 다 획득 시도
    if (std::try_lock(lock1, lock2) == -1) {  // 모두 성공
        // 이체 로직
        return true;
    }
    return false;  // 실패 시 나중에 재시도
}

// 백오프 전략
void transferWithBackoff(Account& from, Account& to, int amount) {
    int backoffMs = 1;
    while (true) {
        if (tryTransfer(from, to, amount)) {
            return;
        }
        // 랜덤 백오프로 동시 재시도 방지
        std::this_thread::sleep_for(
            std::chrono::milliseconds(backoffMs + rand() % backoffMs)
        );
        backoffMs = std::min(backoffMs * 2, 100);  // 최대 100ms
    }
}

Part 3: 데드락 탐지

런타임에 데드락을 탐지할 수 있나요?
Lock 그래프 기반 탐지
// 디버그용 데드락 탐지기
class DeadlockDetector {
private:
    struct LockInfo {
        void* mutex;
        std::thread::id owner;
        std::string location;  // 파일:라인
    };

    std::mutex detectorMutex_;
    std::map heldLocks_;
    std::map> threadLocks_;
    std::map waitingFor_;

public:
    void beforeLock(void* mutex, const char* file, int line) {
        std::lock_guard guard(detectorMutex_);

        auto tid = std::this_thread::get_id();
        waitingFor_[tid] = mutex;

        // 사이클 탐지
        if (detectCycle(tid)) {
            reportDeadlock(tid, mutex, file, line);
        }
    }

    void afterLock(void* mutex, const char* file, int line) {
        std::lock_guard guard(detectorMutex_);

        auto tid = std::this_thread::get_id();
        waitingFor_.erase(tid);

        heldLocks_[mutex] = {mutex, tid, std::string(file) + ":" + std::to_string(line)};
        threadLocks_[tid].push_back(mutex);
    }

    void beforeUnlock(void* mutex) {
        std::lock_guard guard(detectorMutex_);

        auto tid = std::this_thread::get_id();
        heldLocks_.erase(mutex);

        auto& locks = threadLocks_[tid];
        locks.erase(std::remove(locks.begin(), locks.end(), mutex), locks.end());
    }

private:
    bool detectCycle(std::thread::id startTid) {
        std::set visited;
        std::thread::id current = startTid;

        while (true) {
            if (visited.count(current)) {
                return true;  // 사이클 발견!
            }
            visited.insert(current);

            auto it = waitingFor_.find(current);
            if (it == waitingFor_.end()) break;

            auto lockIt = heldLocks_.find(it->second);
            if (lockIt == heldLocks_.end()) break;

            current = lockIt->second.owner;
        }
        return false;
    }

    void reportDeadlock(std::thread::id tid, void* mutex,
                        const char* file, int line) {
        std::cerr << "DEADLOCK DETECTED!\n";
        std::cerr << "Thread " << tid << " waiting for lock at "
                  << file << ":" << line << "\n";
        // 사이클 정보 출력...
        std::abort();  // 또는 예외 throw
    }
};

// 매크로로 사용
#ifdef DEBUG
    #define LOCK(mtx) \
        g_detector.beforeLock(&mtx, __FILE__, __LINE__); \
        mtx.lock(); \
        g_detector.afterLock(&mtx, __FILE__, __LINE__)
    #define UNLOCK(mtx) \
        g_detector.beforeUnlock(&mtx); \
        mtx.unlock()
#else
    #define LOCK(mtx) mtx.lock()
    #define UNLOCK(mtx) mtx.unlock()
#endif

Part 4: 게임에서의 데드락 사례

게임 개발에서 흔한 데드락 패턴은?
사례 1: 이벤트 시스템
// ❌ 데드락 위험
class EventSystem {
    std::mutex mutex_;
    std::vector handlers_;

public:
    void subscribe(Handler h) {
        std::lock_guard lock(mutex_);
        handlers_.push_back(h);
    }

    void fire(Event& e) {
        std::lock_guard lock(mutex_);
        for (auto& h : handlers_) {
            h(e);  // 핸들러가 subscribe() 호출하면 데드락!
        }
    }
};

// ✅ 복사 후 처리
class SafeEventSystem {
    std::mutex mutex_;
    std::vector handlers_;

public:
    void fire(Event& e) {
        std::vector handlersCopy;
        {
            std::lock_guard lock(mutex_);
            handlersCopy = handlers_;  // 복사
        }
        // lock 해제 후 처리
        for (auto& h : handlersCopy) {
            h(e);  // 안전하게 subscribe() 가능
        }
    }
};
사례 2: 계층적 잠금
// 게임 월드 → 엔티티 → 컴포넌트 계층
// ❌ 위험: 다른 순서로 lock 가능
class World {
    std::mutex worldMutex_;
    std::vector entities_;

    void removeEntity(Entity* e) {
        std::lock_guard worldLock(worldMutex_);
        std::lock_guard entityLock(e->mutex_);
        // ...
    }
};

class Entity {
    std::mutex mutex_;
    World* world_;

    void destroy() {
        std::lock_guard entityLock(mutex_);
        std::lock_guard worldLock(world_->mutex_);  // 역순!
        // ...
    }
};

// ✅ 계층 수준(Level) 기반 잠금
enum class LockLevel { World = 0, Entity = 1, Component = 2 };

thread_local LockLevel currentLevel = LockLevel::World;

class HierarchicalMutex {
    std::mutex mutex_;
    LockLevel level_;

public:
    explicit HierarchicalMutex(LockLevel level) : level_(level) {}

    void lock() {
        if (level_ <= currentLevel) {
            throw std::runtime_error("Lock hierarchy violation!");
        }
        mutex_.lock();
        currentLevel = level_;
    }
};

Part 5: 데드락 없는 설계

처음부터 데드락이 안 생기게 설계하려면?
Lock-free 대안
// 싱글 스레드 이벤트 루프 패턴
class GameThread {
    SPSCQueue> taskQueue_;

public:
    // 다른 스레드에서 작업 요청
    void post(std::function task) {
        taskQueue_.push(std::move(task));
    }

    // 메인 루프에서 순차 처리
    void processQueue() {
        std::function task;
        while (taskQueue_.pop(task)) {
            task();  // 단일 스레드에서 실행 → 데드락 불가
        }
    }
};

// 사용 예시
void otherThread() {
    gameThread.post([&] {
        // 게임 스레드에서 안전하게 실행
        entity->damage(10);
    });
}
전략 장점 단점
Lock 순서 고정 간단, 이해 쉬움 모든 코드에서 준수 필요
try_lock + 백오프 유연함 성능 저하 가능
Lock-free 데드락 불가 구현 어려움
단일 스레드 처리 가장 안전 병렬성 감소

📌 핵심 정리

  • 데드락 조건: 상호배제, 점유대기, 비선점, 순환대기 4가지 모두 필요
  • 방지: Lock 순서 고정, std::scoped_lock 사용
  • 탐지: Lock 그래프로 사이클 탐지 (디버그 빌드)
  • 게임 패턴: 이벤트 핸들러에서 lock 피하기, 계층적 잠금
  • 설계: 단일 스레드 이벤트 루프, Lock-free 자료구조 활용
