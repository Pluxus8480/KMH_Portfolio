스레드 우선순위와 친화도는 어떻게 설정하나요?
Windows 스레드 설정
#include 

// 우선순위 설정
void setThreadPriority() {
    HANDLE thread = GetCurrentThread();

    // 우선순위 클래스 (프로세스 단위)
    SetPriorityClass(GetCurrentProcess(), HIGH_PRIORITY_CLASS);

    // 스레드 우선순위
    SetThreadPriority(thread, THREAD_PRIORITY_HIGHEST);
    // THREAD_PRIORITY_TIME_CRITICAL  // 최고 (실시간)
    // THREAD_PRIORITY_HIGHEST        // 높음
    // THREAD_PRIORITY_ABOVE_NORMAL   // 약간 높음
    // THREAD_PRIORITY_NORMAL         // 기본
}

// CPU 친화도 설정 (특정 코어에 고정)
void setThreadAffinity(int coreId) {
    HANDLE thread = GetCurrentThread();
    DWORD_PTR mask = 1ULL << coreId;
    SetThreadAffinityMask(thread, mask);
}

// 오디오 스레드 설정 예시
void setupAudioThread() {
    // 우선순위 최고
    SetThreadPriority(GetCurrentThread(), THREAD_PRIORITY_TIME_CRITICAL);

    // Core 0에 고정 (캐시 효율)
    SetThreadAffinityMask(GetCurrentThread(), 1);

    // MMCSS(Multimedia Class Scheduler Service) 사용
    DWORD taskIndex = 0;
    HANDLE hTask = AvSetMmThreadCharacteristics(TEXT("Pro Audio"), &taskIndex);
}
Linux 스레드 설정
#include 
#include 

// 우선순위 설정
void setThreadPriority() {
    pthread_t thread = pthread_self();

    // 스케줄링 정책 + 우선순위
    struct sched_param param;
    param.sched_priority = 99;  // 1-99 (FIFO/RR)

    pthread_setschedparam(thread, SCHED_FIFO, ¶m);
    // SCHED_FIFO: 실시간, 선점적
    // SCHED_RR: 실시간, Round Robin
    // SCHED_OTHER: 일반 (CFS)
}

// CPU 친화도 설정
void setThreadAffinity(int coreId) {
    cpu_set_t cpuset;
    CPU_ZERO(&cpuset);
    CPU_SET(coreId, &cpuset);

    pthread_setaffinity_np(pthread_self(), sizeof(cpuset), &cpuset);
}

// 게임 스레드 레이아웃 예시
void setupGameThreads() {
    // 메인 스레드: Core 0
    setThreadAffinity(0);

    // 렌더 스레드: Core 1
    std::thread renderThread(renderLoop);
    cpu_set_t cpuset;
    CPU_ZERO(&cpuset);
    CPU_SET(1, &cpuset);
    pthread_setaffinity_np(renderThread.native_handle(),
                          sizeof(cpuset), &cpuset);
}

Part 4: 스레드 풀과 Job System

스레드 풀은 왜 사용하나요?

스레드 생성/소멸 비용을 피하고, 스레드 수를 코어 수에 맞춤

간단한 스레드 풀
class ThreadPool {
private:
    std::vector workers_;
    std::queue> tasks_;
    std::mutex mutex_;
    std::condition_variable condition_;
    bool stop_ = false;

public:
    ThreadPool(size_t numThreads = std::thread::hardware_concurrency()) {
        for (size_t i = 0; i < numThreads; ++i) {
            workers_.emplace_back([this, i] { workerLoop(i); });
        }
    }

    ~ThreadPool() {
        {
            std::lock_guard lock(mutex_);
            stop_ = true;
        }
        condition_.notify_all();
        for (auto& worker : workers_) {
            worker.join();
        }
    }

    template
    auto enqueue(F&& f) -> std::future {
        using ReturnType = decltype(f());

        auto task = std::make_shared>(
            std::forward(f)
        );
        std::future result = task->get_future();

        {
            std::lock_guard lock(mutex_);
            tasks_.emplace([task] { (*task)(); });
        }
        condition_.notify_one();

        return result;
    }

private:
    void workerLoop(int threadId) {
        // 스레드별 친화도 설정
        setThreadAffinity(threadId);

        while (true) {
            std::function task;
            {
                std::unique_lock lock(mutex_);
                condition_.wait(lock, [this] {
                    return stop_ || !tasks_.empty();
                });
                if (stop_ && tasks_.empty()) return;
                task = std::move(tasks_.front());
                tasks_.pop();
            }
            task();
        }
    }
};

스레드 수 = CPU 코어 수 (하이퍼스레딩 제외하면 물리 코어 수)

동기 포인트: Simulate 완료 후 Render 시작 / Render 완료 후 Present / Present은 VSync에 동기화

Fork-Join 패턴
void gameFrame() {
    // === 시뮬레이션 단계 (병렬) ===
    auto aiJob = threadPool.enqueue([&] { updateAI(); });
    auto physicsJob = threadPool.enqueue([&] { updatePhysics(); });
    auto animJob = threadPool.enqueue([&] { updateAnimations(); });

    // 동기 포인트: 모든 시뮬레이션 완료 대기
    aiJob.wait();
    physicsJob.wait();
    animJob.wait();

    // === 렌더 준비 (직렬) ===
    prepareRenderData();  // 시뮬레이션 결과 → 렌더 데이터

    // === 렌더링 단계 (병렬) ===
    auto shadowJob = threadPool.enqueue([&] { renderShadows(); });
    auto opaqueJob = threadPool.enqueue([&] { renderOpaqueObjects(); });

    shadowJob.wait();
    opaqueJob.wait();

    // === 후처리 + 표시 (직렬) ===
    renderPostProcess();
    present();  // VSync 대기
}

📌 핵심 정리

  • 스케줄러: CPU 시간을 프로세스/스레드에 분배
  • 컨텍스트 스위칭: 직접 비용 + 캐시 오염 간접 비용
  • 우선순위: 오디오 등 실시간 스레드에 높은 우선순위 부여
  • 친화도: 특정 코어에 고정하여 캐시 효율 향상
  • 스레드 풀: 스레드 수 = 코어 수, 생성/소멸 비용 제거
