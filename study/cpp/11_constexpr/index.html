constexpr로 컴파일 타임에 계산하기

Part 1: constexpr 기초

constexpr은 무엇인가요?

컴파일 타임에 평가될 수 있는 상수 표현식을 정의합니다!

const vs constexpr
// const: 런타임에 초기화, 이후 변경 불가
const int x = getValue();     // 런타임에 결정
const int y = 10;             // 컴파일 타임에 결정 가능

// constexpr: 반드시 컴파일 타임에 초기화 가능해야 함
constexpr int a = 10;         // OK
// constexpr int b = getValue(); // 에러! (getValue가 constexpr이 아니면)

// 컴파일 타임 vs 런타임 사용 가능 여부
int arr1[y];        // C++에서 허용 (y는 상수 표현식)
int arr2[a];        // OK
int arr3[x];        // 에러! x는 상수 표현식이 아님

// template 인자
std::array<int, a> arr;  // OK
// std::array<int, x> arr2; // 에러!
🎯 비유로 이해하기

요리 레시피(코드)를 작성할 때, 재료 양(값)을 미리 계산해서 적어두면(constexpr) 요리할 때(런타임) 계산할 필요가 없습니다.

Part 2: constexpr 함수

constexpr 함수 정의
// C++11: 단일 return 문만 허용
constexpr int factorial_11(int n) {
    return n <= 1 ? 1 : n * factorial_11(n - 1);
}

// C++14: 일반 함수처럼 작성 가능
constexpr int factorial_14(int n) {
    int result = 1;
    for (int i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

// 컴파일 타임 사용
constexpr int five_fact = factorial_14(5);  // 120, 컴파일 타임
static_assert(five_fact == 120);            // 컴파일 타임 검증

// 런타임에도 사용 가능!
int n;
std::cin >> n;
int result = factorial_14(n);  // 런타임 호출

// constexpr 클래스
class Point {
    int x_, y_;
public:
    constexpr Point(int x, int y) : x_(x), y_(y) {}
    constexpr int x() const { return x_; }
    constexpr int y() const { return y_; }

    constexpr Point operator+(const Point& other) const {
        return Point(x_ + other.x_, y_ + other.y_);
    }
};

constexpr Point p1(1, 2);
constexpr Point p2(3, 4);
constexpr Point p3 = p1 + p2;  // 컴파일 타임!
static_assert(p3.x() == 4 && p3.y() == 6);

Part 3: C++17/20 확장

if constexpr (C++17)
// 컴파일 타임 조건 분기
template<typename T>
auto get_value(T t) {
    if constexpr (std::is_pointer_v<T>) {
        return *t;  // 포인터일 때만 컴파일
    } else {
        return t;   // 그 외
    }
}

int x = 42;
get_value(x);    // 42
get_value(&x);   // 42 (역참조)

// 타입에 따른 다른 처리
template<typename T>
void print(const T& value) {
    if constexpr (std::is_same_v<T, std::string>) {
        std::cout << "String: " << value;
    } else if constexpr (std::is_integral_v<T>) {
        std::cout << "Integer: " << value;
    } else if constexpr (std::is_floating_point_v<T>) {
        std::cout << "Float: " << value;
    } else {
        std::cout << "Unknown type";
    }
}
C++20 constexpr 확장
// constexpr virtual 함수 (C++20)
class Base {
public:
    constexpr virtual int getValue() const { return 1; }
};

class Derived : public Base {
public:
    constexpr int getValue() const override { return 2; }
};

constexpr int test() {
    Derived d;
    const Base& b = d;
    return b.getValue();  // 2, 컴파일 타임!
}

// constexpr new/delete (C++20)
constexpr int* allocTest() {
    int* p = new int(42);
    int value = *p;
    delete p;
    return nullptr;  // 할당/해제가 컴파일 타임 내에서 완료
}

// constexpr std::vector (C++20)
constexpr auto createVector() {
    std::vector<int> v = {1, 2, 3, 4, 5};
    v.push_back(6);
    return v.size();  // 6
}

static_assert(createVector() == 6);

// consteval: 반드시 컴파일 타임에만
consteval int square(int x) {
    return x * x;
}

constexpr int a = square(5);  // OK
// int runtime_value = square(x);  // 에러! 런타임 호출 불가

// constinit: 컴파일 타임 초기화 보장
constinit int global = 42;  // 정적 초기화 순서 문제 방지

Part 4: 컴파일 타임 자료구조

constexpr 배열과 알고리즘
// 컴파일 타임 배열
constexpr std::array<int, 5> arr = {5, 3, 1, 4, 2};

// 컴파일 타임 정렬 (C++20)
constexpr auto sorted = []() {
    auto a = arr;
    std::sort(a.begin(), a.end());
    return a;
}();

static_assert(sorted[0] == 1 && sorted[4] == 5);

// 컴파일 타임 문자열 처리
constexpr size_t strlen_ce(const char* str) {
    size_t len = 0;
    while (str[len]) len++;
    return len;
}

constexpr const char* hello = "Hello, World!";
constexpr size_t len = strlen_ce(hello);  // 13

// 컴파일 타임 해시
constexpr uint32_t fnv1a_32(const char* str) {
    uint32_t hash = 2166136261u;
    while (*str) {
        hash ^= static_cast<uint32_t>(*str++);
        hash *= 16777619u;
    }
    return hash;
}

// 문자열 switch 구현!
void dispatch(const char* command) {
    constexpr uint32_t hash_start = fnv1a_32("start");
    constexpr uint32_t hash_stop = fnv1a_32("stop");
    constexpr uint32_t hash_pause = fnv1a_32("pause");

    switch (fnv1a_32(command)) {
        case hash_start: /* start 처리 */ break;
        case hash_stop:  /* stop 처리 */  break;
        case hash_pause: /* pause 처리 */ break;
    }
}

Part 5: 실전 활용

실용적인 constexpr 예제
// 1. 컴파일 타임 타입 정보
template<typename T>
constexpr const char* type_name() {
    if constexpr (std::is_same_v<T, int>) return "int";
    else if constexpr (std::is_same_v<T, float>) return "float";
    else if constexpr (std::is_same_v<T, double>) return "double";
    else return "unknown";
}

// 2. 컴파일 타임 수학 상수
namespace math {
    constexpr double PI = 3.14159265358979323846;
    constexpr double E = 2.71828182845904523536;

    constexpr double pow(double base, int exp) {
        double result = 1.0;
        for (int i = 0; i < exp; i++) result *= base;
        return result;
    }

    constexpr double sqrt(double x, double curr = 1.0, double prev = 0.0) {
        return curr == prev ? curr
            : sqrt(x, 0.5 * (curr + x / curr), curr);
    }
}

constexpr double sqrt2 = math::sqrt(2.0);

// 3. 컴파일 타임 룩업 테이블
constexpr auto createSinTable() {
    std::array<double, 360> table{};
    for (int i = 0; i < 360; i++) {
        double rad = i * 3.14159265 / 180.0;
        // 테일러 급수로 sin 근사
        double x = rad;
        double term = rad;
        double result = rad;
        for (int n = 1; n < 10; n++) {
            term *= -rad * rad / ((2*n) * (2*n + 1));
            result += term;
        }
        table[i] = result;
    }
    return table;
}

constexpr auto sinTable = createSinTable();

// 4. 컴파일 타임 검증
template<typename T>
constexpr bool isValidConfig() {
    if constexpr (sizeof(T) > 1024) {
        return false;  // 너무 큰 타입
    }
    return true;
}

template<typename Config>
class System {
    static_assert(isValidConfig<Config>(), "Config too large!");
    // ...
};

키워드 평가 시점 런타임 호출
const 런타임 가능 O
constexpr 컴파일/런타임 O
consteval 컴파일만 X
constinit 초기화만 컴파일 O (사용 시)

constexpr auto hugeTable = createHugeTable();  // 컴파일 느려짐

// 복잡한 constexpr은 디버깅 어려움
constexpr int complex() {
    // 에러 메시지가 불친절할 수 있음
}

// 런타임 입력은 constexpr로 처리 불가
int input;
std::cin >> input;
// constexpr int result = factorial(input);  // 에러!

// 해결: 템플릿 사용
template<int N>
struct Factorial {
    static constexpr int value = N * Factorial<N-1>::value;
};

template<>
struct Factorial<0> {
    static constexpr int value = 1;
};

constexpr int fact5 = Factorial<5>::value;  // 120

constexpr은 런타임 비용을 컴파일 타임으로 옮깁니다. 적절히 활용하면 성능 향상!
