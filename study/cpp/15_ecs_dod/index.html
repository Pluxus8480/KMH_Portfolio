Part 1: OOP의 한계

왜 전통적인 OOP가 게임에서 문제가 되나요?

상속 계층이 복잡해지고, 캐시 미스로 성능이 저하됩니다!

OOP 문제점
// 전통적 상속 구조
class GameObject {
    virtual void update() = 0;

class Character : public GameObject {
    virtual void update() override;
    Health health;
    Transform transform;

class FlyingShootingEnemy : public FlyingEnemy, public ShootingEnemy {};

class StaticObject : public GameObject {
    // Transform만 필요한데 다른 것도 상속...

// 문제 3: 캐시 비효율

Part 2: Data-Oriented Design (DOD)

Data-Oriented Design은 어떻게 다른가요?

데이터를 연속적으로 배치하여 캐시 효율을 극대화합니다!

AoS vs SoA
// Array of Structures (AoS) - 전통적 방식
struct Entity {
    Vector3 position;
    Vector3 velocity;
    float health;
    float damage;
};
std::vector<Entity> entities;

// position만 업데이트할 때:
for (auto& e : entities) {
    e.position += e.velocity * dt;
    // 캐시 라인에 health, damage도 로드됨 (낭비!)
}

// Structure of Arrays (SoA) - DOD 방식
struct Entities {
    std::vector<Vector3> positions;   // 연속 배치
    std::vector<Vector3> velocities;  // 연속 배치
    std::vector<float> healths;       // 연속 배치
    std::vector<float> damages;       // 연속 배치
};

// position만 업데이트할 때:
for (size_t i = 0; i < count; i++) {
    positions[i] += velocities[i] * dt;
    // 캐시 라인에 position과 velocity만!
}

SoA 메모리 레이아웃 (캐시 효율적)

positions
pos0pos1pos2pos3pos4pos5
velocities
vel0vel1vel2vel3vel4vel5

캐시 라인 하나에 여러 position! position 순회 시 관련 데이터만 캐시에! SIMD 벡터화도 가능 = 매우 빠름

Part 3: ECS 아키텍처

ECS 핵심 개념
// Entity: 단순한 ID
using Entity = uint32_t;

// Component: 순수 데이터 (로직 없음)
struct Position { float x, y, z; };
struct Velocity { float x, y, z; };
struct Health { float current, max; };
struct Sprite { TextureID texture; int frame; };

// System: 특정 컴포넌트 조합을 처리하는 로직
class MovementSystem {
public:
    void update(float dt) {
        // Position과 Velocity를 가진 모든 엔티티 처리
        for (auto entity : world.view<Position, Velocity>()) {
            auto& pos = world.get<Position>(entity);
            auto& vel = world.get<Velocity>(entity);
            pos.x += vel.x * dt;
            pos.y += vel.y * dt;
            pos.z += vel.z * dt;
        }
    }
};

class RenderSystem {
public:
    void update() {
        // Position과 Sprite를 가진 엔티티만 렌더링
        for (auto entity : world.view<Position, Sprite>()) {
            auto& pos = world.get<Position>(entity);
            auto& sprite = world.get<Sprite>(entity);
            renderer.draw(sprite.texture, pos);
        }
    }
};

// 엔티티 생성
Entity player = world.create();
world.add<Position>(player, {0, 0, 0});
world.add<Velocity>(player, {1, 0, 0});
world.add<Health>(player, {100, 100});
world.add<Sprite>(player, {playerTexture, 0});

Entity bullet = world.create();
world.add<Position>(bullet, {0, 0, 0});
world.add<Velocity>(bullet, {10, 0, 0});
// Health 없음 - 총알은 체력이 필요없음!

Part 4: 간단한 ECS 구현

미니 ECS 프레임워크
// 컴포넌트 타입 ID
inline size_t getComponentTypeId() {
    static size_t id = 0;
    return id++;
}

template<typename T>
size_t getComponentId() {
    static size_t id = getComponentTypeId();
    return id;
}

// 컴포넌트 풀
class ComponentPoolBase {
public:
    virtual ~ComponentPoolBase() = default;
    virtual void remove(Entity e) = 0;
};

template<typename T>
class ComponentPool : public ComponentPoolBase {
    std::unordered_map<Entity, T> data;

public:
    void add(Entity e, T component) {
        data[e] = std::move(component);
    }

    T& get(Entity e) { return data.at(e); }
    bool has(Entity e) const { return data.count(e) > 0; }
    void remove(Entity e) override { data.erase(e); }

    auto begin() { return data.begin(); }
    auto end() { return data.end(); }
};

// World
class World {
    Entity nextEntity = 0;
    std::vector<std::unique_ptr<ComponentPoolBase>> pools;

    template<typename T>
    ComponentPool<T>* getPool() {
        size_t id = getComponentId<T>();
        if (id >= pools.size()) {
            pools.resize(id + 1);
        }
        if (!pools[id]) {
            pools[id] = std::make_unique<ComponentPool<T>>();
        }
        return static_cast<ComponentPool<T>*>(pools[id].get());
    }

public:
    Entity create() { return nextEntity++; }

    template<typename T>
    void add(Entity e, T component) {
        getPool<T>()->add(e, std::move(component));
    }

    template<typename T>
    T& get(Entity e) { return getPool<T>()->get(e); }

    template<typename T>
    bool has(Entity e) { return getPool<T>()->has(e); }

    // 특정 컴포넌트 조합을 가진 엔티티 순회
    template<typename... Components>
    auto view() {
        // 간단 구현: 첫 컴포넌트 풀을 순회하며 필터링
        // 실제로는 더 최적화된 구현 필요
    }
};

Part 5: 성능 최적화 기법

Archetype 기반 ECS
// Archetype: 같은 컴포넌트 조합을 가진 엔티티 그룹
// 장점: 컴포넌트가 연속 배치되어 캐시 효율 극대화

struct Archetype {
    std::vector<ComponentId> componentTypes;

    // 각 컴포넌트 타입의 배열
    std::vector<std::vector<std::byte>> componentArrays;
    std::vector<Entity> entities;
};

// [Position, Velocity] 아키타입
// positions: [pos0, pos1, pos2, pos3, ...]  ← 연속!
// velocities: [vel0, vel1, vel2, vel3, ...]  ← 연속!
// entities: [e0, e1, e2, e3, ...]

// [Position, Velocity, Health] 아키타입
// 다른 테이블에 저장

// SIMD 최적화
void updatePositions(Archetype& arch, float dt) {
    auto* positions = arch.getArray<Position>();
    auto* velocities = arch.getArray<Velocity>();
    size_t count = arch.entities.size();

    // SIMD로 4개씩 처리
    for (size_t i = 0; i + 4 <= count; i += 4) {
        __m128 px = _mm_load_ps(&positions[i].x);
        __m128 vx = _mm_load_ps(&velocities[i].x);
        __m128 dtv = _mm_set1_ps(dt);
        px = _mm_add_ps(px, _mm_mul_ps(vx, dtv));
        _mm_store_ps(&positions[i].x, px);
    }
}
구현 방식 메모리 효율 순회 성능 구현 복잡도
Sparse Set 낮음 중간 낮음
Archetype 높음 높음 높음
Bitset + Pool 중간 중간 중간

Part 6: 실전 ECS 라이브러리

EnTT 예제
#include <entt/entt.hpp>

struct Position { float x, y; };
struct Velocity { float x, y; };
struct Health { int hp; };

int main() {
    entt::registry registry;

    // 엔티티 생성
    auto entity = registry.create();
    registry.emplace<Position>(entity, 0.0f, 0.0f);
    registry.emplace<Velocity>(entity, 1.0f, 0.0f);
    registry.emplace<Health>(entity, 100);

    // 대량 생성
    for (int i = 0; i < 10000; i++) {
        auto e = registry.create();
        registry.emplace<Position>(e, 0.0f, 0.0f);
        registry.emplace<Velocity>(e, rand() % 10, rand() % 10);
    }

    // 시스템: 뷰로 순회
    auto view = registry.view<Position, Velocity>();
    float dt = 1.0f / 60.0f;

    view.each([dt](Position& pos, Velocity& vel) {
        pos.x += vel.x * dt;
        pos.y += vel.y * dt;
    });

    // 그룹: 더 빠른 순회 (정렬된 접근)
    auto group = registry.group<Position>(entt::get<Velocity>);
    for (auto [entity, pos, vel] : group.each()) {
        // 캐시 최적화된 순회
    }

    // 컴포넌트 확인 및 제거
    if (registry.all_of<Health>(entity)) {
        registry.remove<Health>(entity);
    }

    // 엔티티 파괴
    registry.destroy(entity);
}

EnTT, Flecs, EntityX 같은 라이브러리를 사용하면 최적화된 ECS를 바로 활용할 수 있습니다!

📌 핵심 정리

  • DOD: 데이터 배치를 캐시에 최적화
  • SoA: 같은 타입 데이터를 연속 배치
  • Entity: 단순한 ID (데이터나 로직 없음)
  • Component: 순수 데이터 (로직 없음)
  • System: 컴포넌트 조합을 처리하는 로직
  • 장점: 높은 성능, 유연한 조합, 병렬화 용이
