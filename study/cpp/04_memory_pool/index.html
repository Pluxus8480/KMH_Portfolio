Part 1: 기본 할당자의 문제점

new/delete가 왜 느린가요?

시스템 호출, 락, 단편화 등 여러 오버헤드가 있습니다.

  • 시스템 호출: OS 커널에 메모리 요청
  • 락 경쟁: 멀티스레드 환경에서 힙 락
  • 단편화: 다양한 크기 할당/해제로 메모리 조각남
  • 메타데이터: 할당 크기 정보 저장 오버헤드
힙 단편화 문제

초기:  [████████████████████████]
        연속된 메모리

할당:  [A][B][C][D][E][F][G][H]
       다양한 크기 할당

해제:  [A][ ][C][ ][E][ ][G][ ]
       B, D, F, H 해제

문제:  [A][ ][C][ ][E][ ][G][ ]
       빈 공간이 있지만, 큰 블록 할당 불가!
메모리 풀은 어떻게 해결하나요?

미리 큰 메모리 블록을 할당해두고, 작은 조각으로 나눠 재사용합니다.

🎯 비유로 이해하기

매번 종이를 살 때마다 문구점에 가는 대신, 종이 한 박스를 사서 책상에 두고 필요할 때 꺼내 쓰는 것과 같습니다.

Part 2: Fixed-Size Pool

고정 크기 메모리 풀
template<typename T, size_t PoolSize = 1024>
class FixedPool {
private:
    union Block {
        T data;
        Block* next;
    };

    std::array<Block, PoolSize> pool;
    Block* freeList;

public:
    FixedPool() {
        // 프리 리스트 초기화
        freeList = &pool[0];
        for (size_t i = 0; i < PoolSize - 1; i++) {
            pool[i].next = &pool[i + 1];
        }
        pool[PoolSize - 1].next = nullptr;
    }

    T* allocate() {
        if (!freeList) {
            throw std::bad_alloc();
        }

        Block* block = freeList;
        freeList = block->next;
        return reinterpret_cast<T*>(block);
    }

    void deallocate(T* ptr) {
        Block* block = reinterpret_cast<Block*>(ptr);
        block->next = freeList;
        freeList = block;
    }

    template<typename... Args>
    T* construct(Args&&... args) {
        T* ptr = allocate();
        new (ptr) T(std::forward<Args>(args)...);  // placement new
        return ptr;
    }

    void destroy(T* ptr) {
        ptr->~T();  // 소멸자 호출
        deallocate(ptr);
    }
};
Fixed-Size Pool 구조

초기 상태:
freeList → [0] → [1] → [2] → [3] → ... → [N] → null

할당 후:
freeList → [1] → [2] → [3] → ... → [N] → null
           [0] 사용 중

해제 후:
freeList → [0] → [1] → [2] → ... → [N] → null
           [0] 다시 프리 리스트에

Part 3: Object Pool 패턴

객체 재사용 풀
template<typename T>
class ObjectPool {
private:
    struct PoolBlock {
        std::vector<T> objects;
        std::vector<bool> inUse;
        size_t nextFree = 0;
    };

    std::vector<std::unique_ptr<PoolBlock>> blocks;
    size_t blockSize;

    PoolBlock* currentBlock = nullptr;

    void allocateNewBlock() {
        auto block = std::make_unique<PoolBlock>();
        block->objects.resize(blockSize);
        block->inUse.resize(blockSize, false);
        currentBlock = block.get();
        blocks.push_back(std::move(block));
    }

public:
    explicit ObjectPool(size_t blockSize = 64) : blockSize(blockSize) {
        allocateNewBlock();
    }

    T* acquire() {
        // 현재 블록에서 빈 슬롯 찾기
        for (size_t i = currentBlock->nextFree; i < blockSize; i++) {
            if (!currentBlock->inUse[i]) {
                currentBlock->inUse[i] = true;
                currentBlock->nextFree = i + 1;
                return ¤tBlock->objects[i];
            }
        }

        // 다른 블록에서 찾기
        for (auto& block : blocks) {
            for (size_t i = 0; i < blockSize; i++) {
                if (!block->inUse[i]) {
                    block->inUse[i] = true;
                    return &block->objects[i];
                }
            }
        }

        // 새 블록 할당
        allocateNewBlock();
        currentBlock->inUse[0] = true;
        currentBlock->nextFree = 1;
        return ¤tBlock->objects[0];
    }

    void release(T* obj) {
        for (auto& block : blocks) {
            T* begin = block->objects.data();
            T* end = begin + blockSize;
            if (obj >= begin && obj < end) {
                size_t index = obj - begin;
                block->inUse[index] = false;
                if (index < block->nextFree) {
                    block->nextFree = index;
                }
                return;
            }
        }
    }
};

Part 4: STL Allocator 인터페이스

STL 컨테이너에 커스텀 할당자를 적용하려면?
C++17 스타일 Allocator
template<typename T>
class PoolAllocator {
private:
    static inline FixedPool<T, 4096> pool;

public:
    using value_type = T;

    PoolAllocator() noexcept = default;

    template<typename U>
    PoolAllocator(const PoolAllocator<U>&) noexcept {}

    T* allocate(size_t n) {
        if (n != 1) {
            // 여러 개 요청 시 기본 할당자 사용
            return static_cast<T*>(::operator new(n * sizeof(T)));
        }
        return pool.allocate();
    }

    void deallocate(T* p, size_t n) noexcept {
        if (n != 1) {
            ::operator delete(p);
            return;
        }
        pool.deallocate(p);
    }

    template<typename U>
    struct rebind {
        using other = PoolAllocator<U>;
    };
};

template<typename T, typename U>
bool operator==(const PoolAllocator<T>&, const PoolAllocator<U>&) {
    return true;
}

template<typename T, typename U>
bool operator!=(const PoolAllocator<T>&, const PoolAllocator<U>&) {
    return false;
}

// 사용
std::vector<int, PoolAllocator<int>> vec;
std::list<MyObject, PoolAllocator<MyObject>> list;

Part 5: Arena Allocator

대량의 객체를 한 번에 해제하려면?

Arena(Stack) Allocator는 개별 해제 없이 전체를 한 번에 리셋합니다!

Arena Allocator
class ArenaAllocator {
private:
    char* buffer;
    size_t capacity;
    size_t offset = 0;

public:
    explicit ArenaAllocator(size_t size)
        : buffer(new char[size]), capacity(size) {}

    ~ArenaAllocator() {
        delete[] buffer;
    }

    // 정렬된 할당
    void* allocate(size_t size, size_t alignment = alignof(std::max_align_t)) {
        // 정렬 조정
        size_t padding = (alignment - (offset % alignment)) % alignment;
        size_t newOffset = offset + padding + size;

        if (newOffset > capacity) {
            throw std::bad_alloc();
        }

        void* ptr = buffer + offset + padding;
        offset = newOffset;
        return ptr;
    }

    template<typename T, typename... Args>
    T* create(Args&&... args) {
        void* ptr = allocate(sizeof(T), alignof(T));
        return new (ptr) T(std::forward<Args>(args)...);
    }

    // 개별 해제 없음! 전체 리셋만
    void reset() {
        offset = 0;
    }

    size_t used() const { return offset; }
    size_t remaining() const { return capacity - offset; }
};

// 사용 예: 프레임당 임시 객체
class Game {
    ArenaAllocator frameArena{1024 * 1024};  // 1MB

    void update() {
        // 이번 프레임의 임시 객체들
        auto* particles = frameArena.create<Particle[]>(100);
        auto* effects = frameArena.create<Effect[]>(50);

        // 처리...

        // 프레임 끝: 모든 임시 객체 한 번에 해제
        frameArena.reset();
    }
};

벤치마크 예시
void benchmark() {
    const int N = 100000;

    // new/delete
    auto start1 = std::chrono::high_resolution_clock::now();
    for (int i = 0; i < N; i++) {
        auto* p = new MyObject();
        delete p;
    }
    auto end1 = std::chrono::high_resolution_clock::now();

    // Object Pool
    ObjectPool<MyObject> pool;
    auto start2 = std::chrono::high_resolution_clock::now();
    for (int i = 0; i < N; i++) {
        auto* p = pool.acquire();
        pool.release(p);
    }
    auto end2 = std::chrono::high_resolution_clock::now();

    // 결과: Pool이 10~100배 빠름!
}

📌 핵심 정리

  • new/delete 문제: 시스템 호출, 락, 단편화 오버헤드
  • Fixed Pool: Free List로 O(1) 할당/해제
  • Object Pool: 객체 재사용으로 생성자 비용 절감
  • Arena Allocator: 개별 해제 없이 전체 리셋
  • STL Allocator: rebind, allocate/deallocate 인터페이스
