std::optional과 std::variant로 타입 안전하게

Part 1: std::optional이란?

왜 optional이 필요한가요?

"값이 없을 수도 있다"를 표현할 때 nullptr이나 특수값(-1 등)은 위험합니다!

기존 방식의 문제점
// 문제 1: nullptr 반환
User* findUser(int id) {
    // 못 찾으면 nullptr
    return nullptr;
}

User* user = findUser(123);
user->getName();  // 크래시! null 체크 깜빡

// 문제 2: 특수값 사용
int findIndex(const std::vector<int>& vec, int value) {
    // 못 찾으면 -1
    return -1;
}

int idx = findIndex(vec, 42);
vec[idx];  // 문제! -1은 유효한 인덱스 아님

// 문제 3: 출력 매개변수
bool tryParse(const std::string& str, int& result) {
    // 성공하면 true, result에 값 저장
    // 실패하면 false
}

int value;
if (tryParse(input, value)) {
    // value 사용
}
// API가 복잡하고 value가 초기화 안 됐을 수 있음
std::optional 해결책
#include <optional>

std::optional<User> findUser(int id) {
    if (auto it = users.find(id); it != users.end()) {
        return it->second;  // 값 있음
    }
    return std::nullopt;    // 값 없음
}

std::optional<int> findIndex(const std::vector<int>& vec, int value) {
    for (size_t i = 0; i < vec.size(); i++) {
        if (vec[i] == value) return i;
    }
    return std::nullopt;
}

std::optional<int> tryParse(const std::string& str) {
    try {
        return std::stoi(str);
    } catch (...) {
        return std::nullopt;
    }
}

// 사용
if (auto user = findUser(123)) {
    std::cout << user->getName();  // 안전!
}

auto idx = findIndex(vec, 42);
if (idx) {
    vec[*idx];  // 안전!
}
🎯 비유로 이해하기

optional은 선물 상자와 같습니다. 상자 안에 선물이 있을 수도, 없을 수도 있습니다. 열기 전에 먼저 확인하세요!

Part 2: std::optional API

optional 주요 API
std::optional<int> opt;

// 생성
std::optional<int> a;                    // 빈 상태
std::optional<int> b = std::nullopt;     // 빈 상태
std::optional<int> c = 42;               // 값 있음
std::optional<int> d{std::in_place, 42}; // in-place 생성
auto e = std::make_optional(42);         // 편의 함수

// 값 확인
if (opt) { /* 값 있음 */ }
if (opt.has_value()) { /* 값 있음 */ }

// 값 접근
int x = *opt;           // 값 없으면 UB!
int y = opt.value();    // 값 없으면 bad_optional_access 예외
int z = opt.value_or(0); // 값 없으면 기본값

// 값 변경
opt = 100;
opt.emplace(200);      // in-place 생성
opt.reset();           // 빈 상태로

// 비교
std::optional<int> a = 1, b = 2;
a < b;        // true (값 비교)
a < std::nullopt; // false (빈 optional이 가장 작음)

// C++23: monadic 연산
opt.and_then([](int x) -> std::optional<int> { return x * 2; });
opt.transform([](int x) { return x * 2; });
opt.or_else([]() { return std::optional<int>{0}; });
실전 패턴
// 체이닝 패턴
std::optional<User> user = findUser(id);
std::optional<std::string> email =
    user ? std::optional(user->getEmail()) : std::nullopt;

// C++23 스타일 체이닝
auto email = findUser(id)
    .and_then([](User& u) { return u.getProfile(); })
    .transform([](Profile& p) { return p.email; })
    .value_or("no-email@example.com");

// optional 레퍼런스 (직접 지원 안 함)
// 대신 포인터나 reference_wrapper 사용
std::optional<std::reference_wrapper<User>> userRef;

// optional 반환 함수 조합
template<typename T, typename F>
auto flatMap(std::optional<T> opt, F&& f)
    -> decltype(f(*opt)) {
    if (opt) return f(*opt);
    return std::nullopt;
}

Part 3: std::variant란?

variant는 무엇인가요?

타입 안전한 union입니다. 여러 타입 중 하나의 값을 저장할 수 있습니다.

union vs variant
// 기존 union (위험!)
union Value {
    int i;
    float f;
    std::string s;  // 컴파일 에러! 비trivial 타입
};

// 수동 태그 관리
struct TaggedValue {
    enum Type { INT, FLOAT, STRING } type;
    union {
        int i;
        float f;
    };
    std::string s;  // union 밖에 따로
};

// std::variant (안전!)
std::variant<int, float, std::string> value;

value = 42;           // int 저장
value = 3.14f;        // float 저장
value = "hello";      // string 저장

// 현재 타입 확인
if (std::holds_alternative<int>(value)) {
    int i = std::get<int>(value);
}

// 타입 인덱스
size_t idx = value.index();  // 0: int, 1: float, 2: string

Part 4: std::visit 패턴

variant 방문하기
std::variant<int, float, std::string> value = "hello";

// 방법 1: std::get (예외 발생 가능)
try {
    int i = std::get<int>(value);  // bad_variant_access!
} catch (std::bad_variant_access&) {
    // 타입이 다름
}

// 방법 2: std::get_if (nullptr 반환)
if (int* p = std::get_if<int>(&value)) {
    std::cout << *p;
}

// 방법 3: std::visit (가장 권장!)
std::visit([](auto&& arg) {
    using T = std::decay_t<decltype(arg)>;
    if constexpr (std::is_same_v<T, int>) {
        std::cout << "int: " << arg;
    } else if constexpr (std::is_same_v<T, float>) {
        std::cout << "float: " << arg;
    } else {
        std::cout << "string: " << arg;
    }
}, value);

// 오버로드 패턴 (더 깔끔!)
template<class... Ts> struct overloaded : Ts... {
    using Ts::operator()...;
};
template<class... Ts> overloaded(Ts...) -> overloaded<Ts...>;

std::visit(overloaded{
    [](int i)         { std::cout << "int: " << i; },
    [](float f)       { std::cout << "float: " << f; },
    [](std::string& s) { std::cout << "string: " << s; }
}, value);

Part 5: 실전 활용 사례

variant 실전 예제
// 1. 다형성 없는 Shape
struct Circle { float radius; };
struct Rectangle { float width, height; };
struct Triangle { float base, height; };

using Shape = std::variant<Circle, Rectangle, Triangle>;

float area(const Shape& shape) {
    return std::visit(overloaded{
        [](const Circle& c)    { return 3.14159f * c.radius * c.radius; },
        [](const Rectangle& r) { return r.width * r.height; },
        [](const Triangle& t)  { return 0.5f * t.base * t.height; }
    }, shape);
}

std::vector<Shape> shapes = {
    Circle{5.0f},
    Rectangle{3.0f, 4.0f},
    Triangle{6.0f, 8.0f}
};

for (const auto& shape : shapes) {
    std::cout << "Area: " << area(shape) << "\n";
}

// 2. 파서 결과 타입
struct ParseError { std::string message; int line; };
using ParseResult = std::variant<AST, ParseError>;

ParseResult parse(const std::string& code) {
    if (hasError) {
        return ParseError{"Syntax error", 42};
    }
    return buildAST(code);
}

auto result = parse(code);
std::visit(overloaded{
    [](const AST& ast)        { execute(ast); },
    [](const ParseError& err) {
        std::cerr << "Error at line " << err.line << ": " << err.message;
    }
}, result);

// 3. 상태 머신
struct Idle {};
struct Running { int progress; };
struct Completed { std::string result; };
struct Failed { std::string error; };

using State = std::variant<Idle, Running, Completed, Failed>;

class Task {
    State state = Idle{};

public:
    void start() {
        if (std::holds_alternative<Idle>(state)) {
            state = Running{0};
        }
    }

    void update(int progress) {
        if (auto* r = std::get_if<Running>(&state)) {
            r->progress = progress;
            if (progress >= 100) {
                state = Completed{"Done!"};
            }
        }
    }

    void fail(const std::string& error) {
        state = Failed{error};
    }
};

Part 6: 성능과 주의사항

항목 virtual 다형성 std::variant
메모리 힙 할당 + vptr 스택, 가장 큰 타입 크기
호출 비용 vtable 간접 호출 컴파일 타임 분기 (더 빠름)
확장성 새 클래스 추가 쉬움 variant 수정 필요
캐시 포인터 추적 필요 캐시 친화적
주의사항
// 1. variant 크기
std::variant<char, double[100]> v;  // 800+ 바이트!
// 큰 타입은 unique_ptr로 감싸기
std::variant<char, std::unique_ptr<double[100]>> v2;

// 2. valueless_by_exception 상태
std::variant<std::string, ThrowingType> v = "hello";
try {
    v = ThrowingType{};  // 생성자에서 예외 발생
} catch (...) {
    v.valueless_by_exception();  // true! 어떤 값도 없음
}

// 3. monostate로 기본 생성 가능하게
struct NoDefault { NoDefault(int) {} };
// std::variant<NoDefault, int> v;  // 에러! 기본 생성 불가
std::variant<std::monostate, NoDefault, int> v;  // OK

// 4. optional vs variant
// optional<T>는 variant<std::monostate, T>와 유사
// 단일 타입의 유무: optional
// 여러 타입 중 하나: variant

variant는 닫힌 타입 집합에서 가상 함수 없이 다형성을 구현하는 좋은 방법입니다!

📌 핵심 정리

  • optional: 값이 있거나 없는 상태를 안전하게 표현
  • variant: 여러 타입 중 하나를 타입 안전하게 저장
  • std::visit: variant 값에 대한 다형적 처리
  • overloaded 패턴: 람다 오버로딩으로 깔끔한 방문
  • 장점: 힙 할당 없음, 캐시 친화적, 컴파일 타임 타입 체크
