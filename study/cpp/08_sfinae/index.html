Part 1: SFINAE란?

SFINAE가 뭔가요?

Substitution Failure Is Not An Error - 치환 실패는 에러가 아니다!

템플릿 인자 치환이 실패해도 컴파일 에러가 아니라, 그 템플릿만 후보에서 제외됩니다.

SFINAE 기본 원리
// 두 개의 오버로드
template<typename T>
typename T::value_type getValue(T container) {  // #1
    return container[0];
}

template<typename T>
T getValue(T value) {  // #2
    return value;
}

std::vector<int> vec = {1, 2, 3};
int x = 10;

getValue(vec);  // #1 선택: vector에 value_type 있음
getValue(x);    // #2 선택: int에 value_type 없음 → #1 치환 실패 → 제외 → #2 선택
                // 에러가 아님!
🎯 비유로 이해하기

자격증 심사와 같습니다. 지원자 A가 필수 자격증이 없으면 탈락하지만, "자격증 없음"이라고 화내지 않고 다음 지원자를 봅니다.

Part 2: Type Traits 기초

표준 Type Traits
#include <type_traits>

// 타입 검사
std::is_integral_v<int>         // true
std::is_floating_point_v<float> // true
std::is_pointer_v<int*>         // true
std::is_reference_v<int&>        // true
std::is_class_v<std::string>    // true

// 타입 관계
std::is_same_v<int, int32_t>    // true (플랫폼에 따라)
std::is_base_of_v<Base, Derived> // true
std::is_convertible_v<int, double> // true

// 타입 변환
std::remove_const_t<const int>     // int
std::remove_reference_t<int&>       // int
std::add_pointer_t<int>            // int*
std::decay_t<const int&>            // int

// 조건부 타입
std::conditional_t<true, int, float>  // int
std::conditional_t<false, int, float> // float

// 사용 예시
template<typename T>
void print(T value) {
    if constexpr (std::is_integral_v<T>) {
        std::cout << "정수: " << value << "\n";
    } else if constexpr (std::is_floating_point_v<T>) {
        std::cout << "실수: " << value << "\n";
    } else {
        std::cout << "기타: " << value << "\n";
    }
}

Part 3: enable_if 활용

enable_if는 어떻게 동작하나요?
std::enable_if 구현과 사용
// std::enable_if 구현
template<bool Cond, typename T = void>
struct enable_if {};  // 기본: 타입 없음

template<typename T>
struct enable_if<true, T> {
    using type = T;  // 조건이 true면 type 정의
};

// 조건이 false면 enable_if::type이 없음 → 치환 실패 → SFINAE

// 사용법 1: 반환 타입에서
template<typename T>
std::enable_if_t<std::is_integral_v<T>, T>
double_value(T x) {
    return x * 2;
}

template<typename T>
std::enable_if_t<std::is_floating_point_v<T>, T>
double_value(T x) {
    return x * 2.0;
}

// 사용법 2: 템플릿 파라미터에서
template<typename T,
         std::enable_if_t<std::is_integral_v<T>, int> = 0>
void process(T x) {
    std::cout << "정수 처리: " << x << "\n";
}

template<typename T,
         std::enable_if_t<std::is_floating_point_v<T>, int> = 0>
void process(T x) {
    std::cout << "실수 처리: " << x << "\n";
}

// 사용법 3: 클래스 템플릿 특수화
template<typename T, typename Enable = void>
class Serializer;

template<typename T>
class Serializer<T, std::enable_if_t<std::is_arithmetic_v<T>>> {
public:
    static std::string serialize(T value) {
        return std::to_string(value);
    }
};

template<typename T>
class Serializer<T, std::enable_if_t<std::is_same_v<T, std::string>>> {
public:
    static std::string serialize(const T& value) {
        return "\"" + value + "\"";
    }
};

Part 4: 커스텀 Type Traits 만들기

타입 특성 검사기 구현
// 멤버 함수 존재 여부 검사
template<typename T, typename = void>
struct has_size : std::false_type {};

template<typename T>
struct has_size<T, std::void_t<decltype(std::declval<T>().size())>>
    : std::true_type {};

template<typename T>
inline constexpr bool has_size_v = has_size<T>::value;

// 사용
static_assert(has_size_v<std::vector<int>>);  // true
static_assert(!has_size_v<int>);               // true

// 더 정밀한 검사: 반환 타입도 확인
template<typename T, typename = void>
struct has_begin_end : std::false_type {};

template<typename T>
struct has_begin_end<T, std::void_t<
    decltype(std::declval<T>().begin()),
    decltype(std::declval<T>().end())
>> : std::true_type {};

// is_iterable 구현
template<typename T, typename = void>
struct is_iterable : std::false_type {};

template<typename T>
struct is_iterable<T, std::void_t<
    decltype(std::begin(std::declval<T>())),
    decltype(std::end(std::declval<T>()))
>> : std::true_type {};

// is_printable 구현 (operator<< 존재 여부)
template<typename T, typename = void>
struct is_printable : std::false_type {};

template<typename T>
struct is_printable<T, std::void_t<
    decltype(std::declval<std::ostream&>() << std::declval<T>())
>> : std::true_type {};

// 활용
template<typename T>
void smart_print(const T& value) {
    if constexpr (is_iterable<T>::value && !std::is_same_v<T, std::string>) {
        std::cout << "[";
        bool first = true;
        for (const auto& item : value) {
            if (!first) std::cout << ", ";
            smart_print(item);
            first = false;
        }
        std::cout << "]";
    } else if constexpr (is_printable<T>::value) {
        std::cout << value;
    } else {
        std::cout << "(unprintable)";
    }
}

Part 5: 태그 디스패치

SFINAE 외에 다른 방법은?

태그 디스패치: 타입 특성을 태그로 변환하여 오버로딩에 활용!

태그 디스패치 패턴
// 반복자 카테고리별 advance 구현
namespace detail {
    template<typename Iter, typename Distance>
    void advance_impl(Iter& it, Distance n, std::input_iterator_tag) {
        // 입력 반복자: 하나씩 이동
        while (n--) ++it;
    }

    template<typename Iter, typename Distance>
    void advance_impl(Iter& it, Distance n, std::bidirectional_iterator_tag) {
        // 양방향 반복자: 앞뒤로 이동 가능
        if (n >= 0) {
            while (n--) ++it;
        } else {
            while (n++) --it;
        }
    }

    template<typename Iter, typename Distance>
    void advance_impl(Iter& it, Distance n, std::random_access_iterator_tag) {
        // 랜덤 액세스 반복자: 직접 점프
        it += n;
    }
}

template<typename Iter, typename Distance>
void advance(Iter& it, Distance n) {
    // 태그로 디스패치
    detail::advance_impl(it, n,
        typename std::iterator_traits<Iter>::iterator_category{});
}

// 커스텀 태그 사용
struct small_object_tag {};
struct large_object_tag {};

template<typename T>
using object_size_tag = std::conditional_t<
    sizeof(T) <= 16,
    small_object_tag,
    large_object_tag
>;

template<typename T>
void store_impl(T value, small_object_tag) {
    // 작은 객체: 값으로 저장
    buffer_.push_back(value);
}

template<typename T>
void store_impl(T value, large_object_tag) {
    // 큰 객체: 포인터로 저장
    heap_storage_.push_back(std::make_unique<T>(std::move(value)));
}

template<typename T>
void store(T value) {
    store_impl(std::move(value), object_size_tag<T>{});
}

Part 6: 실전 SFINAE 패턴

실전 예제들
// 1. 컨테이너 타입 감지
template<typename T, typename = void>
struct is_container : std::false_type {};

template<typename T>
struct is_container<T, std::void_t<
    typename T::value_type,
    typename T::iterator,
    decltype(std::declval<T>().begin()),
    decltype(std::declval<T>().end()),
    decltype(std::declval<T>().size())
>> : std::true_type {};

// 2. 호출 가능 여부 검사
template<typename F, typename... Args>
struct is_invocable {
private:
    template<typename Func, typename... As>
    static auto test(int) -> decltype(
        std::declval<Func>()(std::declval<As>()...),
        std::true_type{}
    );

    template<typename, typename...>
    static std::false_type test(...);

public:
    static constexpr bool value = decltype(test<F, Args...>(0))::value;
};

// 3. 직렬화 가능 타입 처리
template<typename T, typename = void>
struct has_serialize : std::false_type {};

template<typename T>
struct has_serialize<T, std::void_t<
    decltype(std::declval<T>().serialize())
>> : std::true_type {};

template<typename T>
std::string to_json(const T& obj) {
    if constexpr (has_serialize<T>::value) {
        return obj.serialize();
    } else if constexpr (std::is_arithmetic_v<T>) {
        return std::to_string(obj);
    } else if constexpr (std::is_same_v<T, std::string>) {
        return "\"" + obj + "\"";
    } else if constexpr (is_container<T>::value) {
        std::string result = "[";
        bool first = true;
        for (const auto& item : obj) {
            if (!first) result += ",";
            result += to_json(item);
            first = false;
        }
        return result + "]";
    } else {
        static_assert(sizeof(T) == 0, "Type not serializable");
    }
}

// 4. 생성자 존재 여부 검사
template<typename T, typename... Args>
struct is_constructible_from {
    template<typename U, typename... As>
    static auto test(int) -> decltype(U(std::declval<As>()...), std::true_type{});

    template<typename, typename...>
    static std::false_type test(...);

    static constexpr bool value = decltype(test<T, Args...>(0))::value;
};

📌 핵심 정리

  • SFINAE: 치환 실패는 에러가 아니라 후보 제외
  • enable_if: 조건에 따라 템플릿 활성화/비활성화
  • Type Traits: 컴파일 타임 타입 정보 조회
  • void_t: 멤버 존재 여부 검사에 활용
  • 태그 디스패치: 타입 특성을 오버로딩으로 분기
