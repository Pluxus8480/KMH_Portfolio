자기 자신을 템플릿 인자로?

CRTP로 정적 다형성과 Mixin 구현하기

Part 1: CRTP란?

CRTP가 뭔가요?

파생 클래스가 자기 자신을 기반 클래스의 템플릿 인자로 전달하는 패턴입니다!

CRTP 기본 형태
// CRTP 기반 클래스
template<typename Derived>
class Base {
public:
    void interface() {
        // 파생 클래스의 구현 호출
        static_cast<Derived*>(this)->implementation();
    }

    void defaultImplementation() {
        std::cout << "Base default\n";
    }
};

// 파생 클래스: 자신을 템플릿 인자로 전달
class Derived : public Base<Derived> {  // 핵심!
public:
    void implementation() {
        std::cout << "Derived implementation\n";
    }
};

// 사용
Derived d;
d.interface();  // "Derived implementation" 출력

// 가상 함수 없이 다형성 달성!
🎯 비유로 이해하기

서류 양식에 "이름을 적으세요"라고 써 있으면, 각자 자기 이름을 적습니다. CRTP는 "네 타입을 알려줘"라고 해서 파생 클래스가 자기 타입을 알려주는 것입니다.

Part 2: 정적 다형성

가상 함수 vs CRTP
// 가상 함수 (동적 다형성)
class VirtualShape {
public:
    virtual double area() const = 0;
    virtual ~VirtualShape() = default;
};

class VirtualCircle : public VirtualShape {
    double radius;
public:
    VirtualCircle(double r) : radius(r) {}
    double area() const override { return 3.14159 * radius * radius; }
};

// vtable 조회 비용, 인라인 불가

// CRTP (정적 다형성)
template<typename Derived>
class Shape {
public:
    double area() const {
        return static_cast<const Derived*>(this)->areaImpl();
    }
};

class Circle : public Shape<Circle> {
    double radius;
public:
    Circle(double r) : radius(r) {}
    double areaImpl() const { return 3.14159 * radius * radius; }
};

class Rectangle : public Shape<Rectangle> {
    double width, height;
public:
    Rectangle(double w, double h) : width(w), height(h) {}
    double areaImpl() const { return width * height; }
};

// 컴파일 타임에 타입 결정, 인라인 가능!

// 템플릿 함수에서 사용
template<typename T>
void printArea(const Shape<T>& shape) {
    std::cout << "Area: " << shape.area() << "\n";
}

Circle c(5.0);
Rectangle r(3.0, 4.0);
printArea(c);  // 인라인됨!
printArea(r);  // 인라인됨!
항목 가상 함수 CRTP
다형성 시점 런타임 컴파일타임
vtable 오버헤드 있음 없음
인라인 어려움 가능
이질적 컬렉션 가능 불가

Part 3: Mixin 패턴

CRTP로 기능을 추가할 수 있나요?

Mixin: 상속을 통해 기능을 "믹스인"하는 패턴!

CRTP Mixin
// 비교 연산 Mixin
template<typename Derived>
class Comparable {
public:
    friend bool operator!=(const Derived& a, const Derived& b) {
        return !(a == b);  // Derived가 == 제공
    }

    friend bool operator>(const Derived& a, const Derived& b) {
        return b < a;  // Derived가 < 제공
    }

    friend bool operator<=(const Derived& a, const Derived& b) {
        return !(b < a);
    }

    friend bool operator>=(const Derived& a, const Derived& b) {
        return !(a < b);
    }
};

class Point : public Comparable<Point> {
    int x, y;
public:
    Point(int x, int y) : x(x), y(y) {}

    // ==와 <만 구현
    bool operator==(const Point& other) const {
        return x == other.x && y == other.y;
    }

    bool operator<(const Point& other) const {
        return std::tie(x, y) < std::tie(other.x, other.y);
    }
};

// 자동으로 !=, >, <=, >= 사용 가능!
Point p1{1, 2}, p2{3, 4};
if (p1 < p2) { /* ... */ }
if (p1 != p2) { /* ... */ }
if (p1 >= p2) { /* ... */ }

Part 4: 실전 활용

싱글턴 CRTP
// 싱글턴 Mixin
template<typename Derived>
class Singleton {
protected:
    Singleton() = default;
    ~Singleton() = default;

public:
    Singleton(const Singleton&) = delete;
    Singleton& operator=(const Singleton&) = delete;

    static Derived& getInstance() {
        static Derived instance;
        return instance;
    }
};

class Logger : public Singleton<Logger> {
    friend class Singleton<Logger>;  // 생성자 접근 허용
    Logger() = default;

public:
    void log(const std::string& msg) {
        std::cout << msg << "\n";
    }
};

// 사용
Logger::getInstance().log("Hello!");
Counter Mixin
// 객체 수 추적
template<typename Derived>
class Counter {
    static inline size_t count = 0;

protected:
    Counter() { ++count; }
    Counter(const Counter&) { ++count; }
    ~Counter() { --count; }

public:
    static size_t getCount() { return count; }
};

class Enemy : public Counter<Enemy> {
    // ...
};

class Bullet : public Counter<Bullet> {
    // ...
};

// 사용
void gameLoop() {
    std::cout << "Enemies: " << Enemy::getCount() << "\n";
    std::cout << "Bullets: " << Bullet::getCount() << "\n";
}
Cloneable 패턴
// 다형적 복사
template<typename Derived>
class Cloneable {
public:
    std::unique_ptr<Derived> clone() const {
        return std::make_unique<Derived>(
            static_cast<const Derived&>(*this)
        );
    }
};

class Document : public Cloneable<Document> {
    std::string content;
public:
    Document(const std::string& c) : content(c) {}
    Document(const Document&) = default;
};

// 사용
Document doc("Hello");
auto copy = doc.clone();  // 타입 안전한 복사!

Part 5: Chaining과 Builder

메서드 체이닝
// CRTP로 반환 타입을 파생 클래스로
template<typename Derived>
class Builder {
protected:
    Derived& self() { return static_cast<Derived&>(*this); }

public:
    Derived& setName(const std::string& name) {
        // 설정...
        return self();
    }

    Derived& setValue(int value) {
        // 설정...
        return self();
    }
};

class AdvancedBuilder : public Builder<AdvancedBuilder> {
public:
    AdvancedBuilder& setAdvancedOption(bool opt) {
        // 설정...
        return *this;
    }
};

// 체이닝 가능!
AdvancedBuilder builder;
builder
    .setName("Test")        // Builder 메서드
    .setValue(42)           // Builder 메서드
    .setAdvancedOption(true); // AdvancedBuilder 메서드

// Base::setName이 Base& 반환하면 체이닝 끊김
// CRTP로 Derived& 반환하여 해결!

Part 6: 주의사항

CRTP 함정들
// 1. 잘못된 파생 클래스 전달
class Wrong : public Base<OtherClass> {  // 버그!
    // ...
};

// 방지책: C++23 deducing this 또는 체크
template<typename Derived>
class Base {
    Base() {
        static_assert(std::is_base_of_v<Base, Derived>,
                      "Invalid CRTP usage");
    }
};

// 2. 불완전 타입 문제
template<typename Derived>
class Base {
    void foo() {
        // Derived가 아직 불완전할 수 있음
        sizeof(Derived);  // 일부 컨텍스트에서 에러
    }
};

// 3. 다이아몬드 상속
class A : public Mixin1<A>, public Mixin2<A> {
    // 두 Mixin이 같은 Base를 상속하면 문제
};

// 4. 코드 비대화
// 각 파생 클래스마다 Base 코드가 인스턴스화됨

// C++23 대안: deducing this
class Modern {
public:
    // this 타입이 자동 추론됨
    template<typename Self>
    Self& setName(this Self& self, std::string name) {
        // ...
        return self;
    }
};

CRTP는 강력하지만 복잡합니다. C++23의 deducing this가 많은 경우를 대체할 수 있습니다!

📌 핵심 정리

  • CRTP: class Derived : public Base<Derived>
  • 정적 다형성: vtable 없이 다형적 동작
  • Mixin: 공통 기능을 상속으로 추가
  • 체이닝: 반환 타입을 파생 클래스로
  • 활용: Singleton, Counter, Comparable, Cloneable
