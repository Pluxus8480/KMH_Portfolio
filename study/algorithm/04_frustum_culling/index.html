Part 2: 절두체 평면 추출

절두체의 6개 평면을 어떻게 구하나요?

View-Projection 행렬에서 직접 추출할 수 있습니다!

C++ 절두체 평면 추출
struct Plane {
    float a, b, c, d;  // ax + by + cz + d = 0

    void normalize() {
        float len = std::sqrt(a*a + b*b + c*c);
        a /= len; b /= len; c /= len; d /= len;
    }

    // 점과 평면의 부호 거리
    float distance(float x, float y, float z) const {
        return a * x + b * y + c * z + d;
    }
};

class Frustum {
public:
    Plane planes[6];  // Left, Right, Bottom, Top, Near, Far

    // View-Projection 행렬에서 절두체 추출
    void extractFromMatrix(const float* vp) {
        // vp는 4x4 행렬 (column-major)
        // m[row][col] = vp[col * 4 + row]

        // Left: row3 + row0
        planes[0] = {vp[3] + vp[0], vp[7] + vp[4],
                     vp[11] + vp[8], vp[15] + vp[12]};

        // Right: row3 - row0
        planes[1] = {vp[3] - vp[0], vp[7] - vp[4],
                     vp[11] - vp[8], vp[15] - vp[12]};

        // Bottom: row3 + row1
        planes[2] = {vp[3] + vp[1], vp[7] + vp[5],
                     vp[11] + vp[9], vp[15] + vp[13]};

        // Top: row3 - row1
        planes[3] = {vp[3] - vp[1], vp[7] - vp[5],
                     vp[11] - vp[9], vp[15] - vp[13]};

        // Near: row3 + row2
        planes[4] = {vp[3] + vp[2], vp[7] + vp[6],
                     vp[11] + vp[10], vp[15] + vp[14]};

        // Far: row3 - row2
        planes[5] = {vp[3] - vp[2], vp[7] - vp[6],
                     vp[11] - vp[10], vp[15] - vp[14]};

        // 정규화
        for (int i = 0; i < 6; i++) {
            planes[i].normalize();
        }
    }
};

Part 3: AABB vs 절두체 판정

물체가 절두체 안에 있는지 어떻게 검사하나요?

물체의 바운딩 박스(AABB)를 6개 평면과 비교합니다.

Inside
모든 점이 평면 앞
Intersect
일부 점이 평면 뒤
Outside
모든 점이 평면 뒤

최적화된 판정 방법:

  • 각 평면에 대해 "가장 가까운 점"만 검사
  • 그 점이 평면 뒤에 있으면 완전히 밖
최적화된 AABB-절두체 검사
struct AABB {
    float minX, minY, minZ;
    float maxX, maxY, maxZ;

    // 평면 법선 방향으로 가장 먼 점 (P-vertex)
    Vec3 getPositiveVertex(const Plane& plane) const {
        return {
            plane.a >= 0 ? maxX : minX,
            plane.b >= 0 ? maxY : minY,
            plane.c >= 0 ? maxZ : minZ
        };
    }

    // 평면 법선 반대 방향으로 가장 먼 점 (N-vertex)
    Vec3 getNegativeVertex(const Plane& plane) const {
        return {
            plane.a >= 0 ? minX : maxX,
            plane.b >= 0 ? minY : maxY,
            plane.c >= 0 ? minZ : maxZ
        };
    }
};

enum class CullResult {
    Inside,      // 완전히 내부
    Outside,     // 완전히 외부
    Intersect    // 경계에 걸침
};

CullResult frustumCullAABB(const Frustum& frustum, const AABB& box) {
    CullResult result = CullResult::Inside;

    for (int i = 0; i < 6; i++) {
        const Plane& plane = frustum.planes[i];

        // N-vertex가 평면 앞에 있으면 완전히 내부
        Vec3 nVertex = box.getNegativeVertex(plane);
        if (plane.distance(nVertex.x, nVertex.y, nVertex.z) > 0) {
            // P-vertex 검사 불필요
        }
        else {
            // P-vertex가 평면 뒤에 있으면 완전히 외부
            Vec3 pVertex = box.getPositiveVertex(plane);
            if (plane.distance(pVertex.x, pVertex.y, pVertex.z) < 0) {
                return CullResult::Outside;  // 컬링!
            }
            result = CullResult::Intersect;
        }
    }

    return result;
}

Part 4: 구(Sphere)를 이용한 빠른 검사

더 빠른 방법은 없나요?

바운딩 스피어를 먼저 검사하면 더 빠릅니다!

바운딩 스피어 절두체 검사
struct BoundingSphere {
    float x, y, z;   // 중심
    float radius;
};

bool frustumCullSphere(const Frustum& frustum, const BoundingSphere& sphere) {
    for (int i = 0; i < 6; i++) {
        const Plane& plane = frustum.planes[i];

        // 중심과 평면의 거리
        float dist = plane.distance(sphere.x, sphere.y, sphere.z);

        // 구 전체가 평면 뒤에 있으면 컬링
        if (dist < -sphere.radius) {
            return true;  // 컬링됨!
        }
    }

    return false;  // 보임
}
바운딩 볼륨 연산량 정확도
스피어 6번 내적 낮음 (여유 공간)
AABB 12~24번 연산 중간
OBB 더 많음 높음

스피어로 빠르게 걸러내고, 경계에 걸치면 AABB로 정밀 검사!

Part 5: 계층적 컬링 시스템

수천 개의 물체를 효율적으로 컬링하려면?

공간 분할 구조와 결합하여 계층적으로 컬링합니다.

루트
전체 밖 → 스킵!
전체 안 → 자식 검사
일부 안 → 더 검사
일부 밖 → 스킵!

Octree + Frustum Culling 계층적 컬링

Octree + Frustum Culling
class OctreeFrustumCuller {
public:
    void cullNode(OctreeNode* node, const Frustum& frustum,
                  std::vector<GameObject*>& visible) {
        // 1. 노드 바운딩 박스 검사
        CullResult result = frustumCullAABB(frustum, node->bounds);

        if (result == CullResult::Outside) {
            return;  // 전체 노드 컬링!
        }

        if (result == CullResult::Inside) {
            // 전체 노드가 절두체 안 - 모든 물체 추가
            addAllObjects(node, visible);
            return;
        }

        // Intersect - 자식 노드 또는 개별 물체 검사
        if (node->isLeaf()) {
            for (auto* obj : node->objects) {
                if (!frustumCullSphere(frustum, obj->boundingSphere)) {
                    visible.push_back(obj);
                }
            }
        } else {
            for (int i = 0; i < 8; i++) {
                if (node->children[i]) {
                    cullNode(node->children[i], frustum, visible);
                }
            }
        }
    }

private:
    void addAllObjects(OctreeNode* node, std::vector<GameObject*>& visible) {
        for (auto* obj : node->objects) {
            visible.push_back(obj);
        }
        if (!node->isLeaf()) {
            for (int i = 0; i < 8; i++) {
                if (node->children[i]) {
                    addAllObjects(node->children[i], visible);
                }
            }
        }
    }
};

Part 6: GPU 기반 컬링

CPU 컬링의 한계는?

물체 수가 매우 많으면 CPU 컬링도 병목이 됩니다.

GPU Compute Shader로 병렬 컬링이 가능합니다!

HLSL GPU Frustum Culling
// ComputeShader
[numthreads(64, 1, 1)]
void FrustumCull(uint3 id : SV_DispatchThreadID)
{
    uint instanceIndex = id.x;
    if (instanceIndex >= instanceCount) return;

    // 인스턴스 바운딩 스피어
    float4 boundingSphere = instanceData[instanceIndex].boundingSphere;
    float3 center = boundingSphere.xyz;
    float radius = boundingSphere.w;

    // 6개 평면 테스트
    bool visible = true;
    for (int i = 0; i < 6; i++)
    {
        float dist = dot(frustumPlanes[i].xyz, center) + frustumPlanes[i].w;
        if (dist < -radius)
        {
            visible = false;
            break;
        }
    }

    // Indirect Draw Buffer에 가시 인스턴스만 추가
    if (visible)
    {
        uint index;
        InterlockedAdd(visibleCount[0], 1, index);
        visibleInstances[index] = instanceIndex;
    }
}

GPU 컬링 + Indirect Drawing = 수십만 물체도 빠르게 처리!
