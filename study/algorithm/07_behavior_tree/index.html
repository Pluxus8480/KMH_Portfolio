Part 1: FSM에서 Behavior Tree로

기존 FSM(상태 기계)의 문제점은 무엇인가요?

상태가 많아지면 전이(Transition)가 기하급수적으로 증가합니다.

Idle
↔
Patrol
Chase
↔
Attack

FSM의 문제: 상태 5개 → 최대 20개 전이 / 상태 10개 → 최대 90개 전이!

Behavior Tree는 행동을 트리 구조로 조직화하여 이 문제를 해결합니다.

Behavior Tree의 장점은 무엇인가요?

Part 2: Behavior Tree 기본 구조

Behavior Tree는 어떤 노드로 구성되나요?

노드 타입:

  • Action (Leaf): 실제 행동 실행
  • Condition (Leaf): 조건 검사
  • Composite: 자식 노드 제어 (Selector, Sequence)
  • Decorator: 자식 노드 수정 (반복, 반전 등)

반환 상태:

  • Success: 작업 성공
  • Failure: 작업 실패
  • Running: 작업 진행 중

Selector: 왼쪽부터 시도, 하나 성공하면 종료 / Sequence: 왼쪽부터 실행, 하나 실패하면 종료

Part 3: Composite 노드 구현

C++ Behavior Tree 기본 구조
#include <vector>
#include <memory>

enum class NodeStatus {
    Success,
    Failure,
    Running
};

// 기본 노드 인터페이스
class BTNode {
public:
    virtual ~BTNode() = default;
    virtual NodeStatus tick(float deltaTime) = 0;
    virtual void reset() {}
};

using BTNodePtr = std::shared_ptr<BTNode>;

// Selector: OR 로직 - 하나라도 성공하면 성공
class Selector : public BTNode {
private:
    std::vector<BTNodePtr> children;
    int currentChild = 0;

public:
    void addChild(BTNodePtr child) {
        children.push_back(child);
    }

    NodeStatus tick(float deltaTime) override {
        while (currentChild < children.size()) {
            NodeStatus status = children[currentChild]->tick(deltaTime);

            if (status == NodeStatus::Running) {
                return NodeStatus::Running;
            }

            if (status == NodeStatus::Success) {
                reset();
                return NodeStatus::Success;  // 하나 성공 = 전체 성공
            }

            // Failure: 다음 자식 시도
            currentChild++;
        }

        reset();
        return NodeStatus::Failure;  // 모두 실패
    }

    void reset() override {
        currentChild = 0;
        for (auto& child : children) {
            child->reset();
        }
    }
};

// Sequence: AND 로직 - 모두 성공해야 성공
class Sequence : public BTNode {
private:
    std::vector<BTNodePtr> children;
    int currentChild = 0;

public:
    void addChild(BTNodePtr child) {
        children.push_back(child);
    }

    NodeStatus tick(float deltaTime) override {
        while (currentChild < children.size()) {
            NodeStatus status = children[currentChild]->tick(deltaTime);

            if (status == NodeStatus::Running) {
                return NodeStatus::Running;
            }

            if (status == NodeStatus::Failure) {
                reset();
                return NodeStatus::Failure;  // 하나 실패 = 전체 실패
            }

            // Success: 다음 자식으로
            currentChild++;
        }

        reset();
        return NodeStatus::Success;  // 모두 성공
    }

    void reset() override {
        currentChild = 0;
        for (auto& child : children) {
            child->reset();
        }
    }
};

Part 4: Action과 Condition 노드

Action과 Condition 예시
// Blackboard: AI 상태 공유 데이터
class Blackboard {
public:
    std::unordered_map<std::string, std::any> data;

    template<typename T>
    void set(const std::string& key, const T& value) {
        data[key] = value;
    }

    template<typename T>
    T get(const std::string& key) const {
        return std::any_cast<T>(data.at(key));
    }

    bool has(const std::string& key) const {
        return data.find(key) != data.end();
    }
};

// Condition: 적이 범위 내에 있는가?
class IsEnemyInRange : public BTNode {
private:
    Blackboard* blackboard;
    float range;

public:
    IsEnemyInRange(Blackboard* bb, float r) : blackboard(bb), range(r) {}

    NodeStatus tick(float deltaTime) override {
        if (!blackboard->has("enemy")) {
            return NodeStatus::Failure;
        }

        auto self = blackboard->get<Vec3>("selfPosition");
        auto enemy = blackboard->get<Vec3>("enemyPosition");
        float distance = (enemy - self).length();

        return distance <= range ? NodeStatus::Success : NodeStatus::Failure;
    }
};

// Action: 적에게 이동
class MoveToEnemy : public BTNode {
private:
    Blackboard* blackboard;
    float speed;
    float arrivalThreshold = 1.0f;

public:
    MoveToEnemy(Blackboard* bb, float s) : blackboard(bb), speed(s) {}

    NodeStatus tick(float deltaTime) override {
        auto& self = blackboard->get<Vec3>("selfPosition");
        auto enemy = blackboard->get<Vec3>("enemyPosition");

        Vec3 direction = (enemy - self).normalized();
        self = self + direction * speed * deltaTime;

        float distance = (enemy - self).length();
        if (distance <= arrivalThreshold) {
            return NodeStatus::Success;  // 도착
        }

        return NodeStatus::Running;  // 이동 중
    }
};

// Action: 공격
class AttackEnemy : public BTNode {
private:
    Blackboard* blackboard;
    float attackCooldown;
    float timer = 0;

public:
    AttackEnemy(Blackboard* bb, float cooldown)
        : blackboard(bb), attackCooldown(cooldown) {}

    NodeStatus tick(float deltaTime) override {
        timer += deltaTime;

        if (timer >= attackCooldown) {
            // 공격 실행
            auto enemy = blackboard->get<Entity*>("enemy");
            enemy->takeDamage(10);

            timer = 0;
            return NodeStatus::Success;
        }

        return NodeStatus::Running;  // 쿨다운 중
    }

    void reset() override {
        timer = 0;
    }
};

Part 5: Decorator 노드

Decorator는 어떤 역할을 하나요?

자식 노드의 결과를 수정하거나 실행 방식을 변경합니다.

Decorator 구현
// Inverter: 결과 반전 (Success ↔ Failure)
class Inverter : public BTNode {
private:
    BTNodePtr child;

public:
    Inverter(BTNodePtr c) : child(c) {}

    NodeStatus tick(float deltaTime) override {
        NodeStatus status = child->tick(deltaTime);

        if (status == NodeStatus::Success) return NodeStatus::Failure;
        if (status == NodeStatus::Failure) return NodeStatus::Success;
        return NodeStatus::Running;
    }
};

// Repeater: N번 반복
class Repeater : public BTNode {
private:
    BTNodePtr child;
    int repeatCount;
    int currentCount = 0;

public:
    Repeater(BTNodePtr c, int count) : child(c), repeatCount(count) {}

    NodeStatus tick(float deltaTime) override {
        if (currentCount >= repeatCount) {
            reset();
            return NodeStatus::Success;
        }

        NodeStatus status = child->tick(deltaTime);

        if (status == NodeStatus::Success) {
            currentCount++;
            child->reset();
        } else if (status == NodeStatus::Failure) {
            reset();
            return NodeStatus::Failure;
        }

        return NodeStatus::Running;
    }

    void reset() override {
        currentCount = 0;
        child->reset();
    }
};

// UntilFail: 자식이 실패할 때까지 반복
class UntilFail : public BTNode {
private:
    BTNodePtr child;

public:
    UntilFail(BTNodePtr c) : child(c) {}

    NodeStatus tick(float deltaTime) override {
        NodeStatus status = child->tick(deltaTime);

        if (status == NodeStatus::Failure) {
            return NodeStatus::Success;  // 실패하면 성공 반환
        }

        if (status == NodeStatus::Success) {
            child->reset();  // 성공하면 다시 실행
        }

        return NodeStatus::Running;
    }
};

// Cooldown: 쿨다운 적용
class Cooldown : public BTNode {
private:
    BTNodePtr child;
    float cooldownTime;
    float timer = 0;
    bool onCooldown = false;

public:
    Cooldown(BTNodePtr c, float cd) : child(c), cooldownTime(cd) {}

    NodeStatus tick(float deltaTime) override {
        if (onCooldown) {
            timer += deltaTime;
            if (timer >= cooldownTime) {
                timer = 0;
                onCooldown = false;
            } else {
                return NodeStatus::Failure;  // 쿨다운 중
            }
        }

        NodeStatus status = child->tick(deltaTime);

        if (status != NodeStatus::Running) {
            onCooldown = true;
        }

        return status;
    }
};

class EnemyAI {
private:
    BTNodePtr rootNode;
    Blackboard blackboard;

public:
    void buildBehaviorTree() {
        // Combat 서브트리
        auto attack = std::make_shared<Sequence>();
        attack->addChild(std::make_shared<IsEnemyInRange>(&blackboard, 2.0f));
        attack->addChild(std::make_shared<Cooldown>(
            std::make_shared<AttackEnemy>(&blackboard, 0.5f), 1.0f));

        auto chase = std::make_shared<Sequence>();
        chase->addChild(std::make_shared<IsEnemyVisible>(&blackboard));
        chase->addChild(std::make_shared<MoveToEnemy>(&blackboard, 5.0f));

        auto combat = std::make_shared<Selector>();
        combat->addChild(attack);
        combat->addChild(chase);

        // Investigate 서브트리
        auto investigate = std::make_shared<Sequence>();
        investigate->addChild(std::make_shared<HasLastKnownPosition>(&blackboard));
        investigate->addChild(std::make_shared<MoveToPosition>(&blackboard, "lastKnownPos", 3.0f));
        investigate->addChild(std::make_shared<LookAround>(&blackboard, 2.0f));

        // Patrol 서브트리
        auto patrol = std::make_shared<Sequence>();
        patrol->addChild(std::make_shared<GetNextPatrolPoint>(&blackboard));
        patrol->addChild(std::make_shared<MoveToPosition>(&blackboard, "patrolTarget", 2.0f));
        patrol->addChild(std::make_shared<Wait>(2.0f));

        // 루트
        auto root = std::make_shared<Selector>();
        root->addChild(combat);
        root->addChild(investigate);
        root->addChild(std::make_shared<UntilFail>(patrol));

        rootNode = root;
    }

    void update(float deltaTime) {
        // 센서 데이터 업데이트
        updateBlackboard();

        // 트리 실행
        rootNode->tick(deltaTime);
    }

    void updateBlackboard() {
        blackboard.set("selfPosition", getPosition());

        Entity* nearestEnemy = findNearestEnemy();
        if (nearestEnemy && canSee(nearestEnemy)) {
            blackboard.set("enemy", nearestEnemy);
            blackboard.set("enemyPosition", nearestEnemy->getPosition());
            blackboard.set("lastKnownPos", nearestEnemy->getPosition());
        } else {
            blackboard.data.erase("enemy");
            blackboard.data.erase("enemyPosition");
        }
    }
};

📌 핵심 정리

  • Behavior Tree: 계층적 트리 구조로 AI 행동 조직화
  • Selector: OR 로직, 하나라도 성공하면 성공
  • Sequence: AND 로직, 모두 성공해야 성공
  • Decorator: 자식 노드의 결과/실행 방식 수정
  • Blackboard: AI 상태 데이터 공유 저장소
  • 장점: 모듈성, 재사용성, 디버깅 용이
