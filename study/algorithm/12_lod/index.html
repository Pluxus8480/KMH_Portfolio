Part 1: LOD의 필요성

왜 LOD가 필요한가요?

멀리 있는 물체는 화면에서 몇 픽셀만 차지합니다.

10만 폴리곤 모델이 화면에 10픽셀로 그려진다면? → 99.99% 낭비!

LOD 0 (가까움)
50,000 폴리곤
화면 500px
→
LOD 1 (중간)
5,000 폴리곤
화면 50px
→
LOD 2 (멀리)
500 폴리곤
화면 5px
어떤 기준으로 LOD를 선택하나요?

거리 또는 화면 점유율(Screen Coverage)을 기준으로 선택합니다.

🎯 비유로 이해하기

멀리서 볼 때는 숲 전체를 보지만, 가까이 가면 나무 한 그루의 나뭇잎이 보입니다.

거리에 따라 디테일 수준을 조절하는 것이 LOD입니다.

Part 2: 거리 기반 LOD

기본 거리 기반 LOD
struct LODLevel {
    float maxDistance;      // 이 LOD가 사용되는 최대 거리
    Mesh* mesh;             // 해당 레벨의 메시
    int triangleCount;      // 삼각형 수 (디버깅용)
};

class LODObject {
private:
    std::vector<LODLevel> levels;
    Vec3 position;
    float boundingRadius;

public:
    void setupLOD() {
        // LOD 레벨 설정 (거리순 정렬)
        levels = {
            {50.0f,  highDetailMesh,   50000},   // LOD 0: 50m 이내
            {150.0f, mediumDetailMesh, 10000},   // LOD 1: 50~150m
            {500.0f, lowDetailMesh,    2000},    // LOD 2: 150~500m
            {FLT_MAX, billboardMesh,   2}        // LOD 3: 500m 이상 (빌보드)
        };
    }

    int selectLOD(const Vec3& cameraPos) const {
        float distance = (cameraPos - position).length();

        for (int i = 0; i < levels.size(); i++) {
            if (distance <= levels[i].maxDistance) {
                return i;
            }
        }
        return levels.size() - 1;
    }

    Mesh* getMeshForLOD(const Vec3& cameraPos) const {
        int lodIndex = selectLOD(cameraPos);
        return levels[lodIndex].mesh;
    }
};

Part 3: 화면 점유율 기반 LOD

거리만으로 충분하지 않나요?

큰 물체와 작은 물체가 같은 거리에 있어도 화면 점유율은 다릅니다.

화면 점유율 = 물체가 차지하는 픽셀 수 / 전체 픽셀

화면 점유율 기반 LOD
class ScreenCoverageLOD {
private:
    float screenWidth, screenHeight;
    float fov;  // Field of View (라디안)

public:
    // 화면에서 물체가 차지하는 대략적 높이 (픽셀)
    float computeScreenSize(float objectRadius, float distance) const {
        // 투영된 크기 계산
        float projectedSize = objectRadius / (distance * tan(fov * 0.5f));
        return projectedSize * screenHeight;
    }

    int selectLOD(float objectRadius, float distance,
                  const std::vector<float>& thresholds) const {
        float screenSize = computeScreenSize(objectRadius, distance);

        // 화면 크기가 클수록 높은 LOD 필요
        for (int i = 0; i < thresholds.size(); i++) {
            if (screenSize >= thresholds[i]) {
                return i;  // LOD 0이 가장 상세
            }
        }
        return thresholds.size();  // 가장 낮은 LOD
    }
};

// 사용 예
ScreenCoverageLOD lodSelector(1920, 1080, glm::radians(60.0f));

// 화면에서 100픽셀 이상 → LOD 0
// 50~100픽셀 → LOD 1
// 50픽셀 미만 → LOD 2
std::vector<float> thresholds = {100.0f, 50.0f, 20.0f};

int lod = lodSelector.selectLOD(object.radius, distanceToCamera, thresholds);

Part 4: LOD 전환 기법

LOD 전환이 눈에 띄지 않게 하려면?

갑작스러운 LOD 전환은 "팝핑(Popping)" 현상을 유발합니다.

해결책 1: 히스테리시스(Hysteresis)

히스테리시스 적용
class LODWithHysteresis {
private:
    int currentLOD = 0;
    float hysteresis = 0.1f;  // 10% 마진

public:
    int selectLOD(float distance, const std::vector<float>& thresholds) {
        int newLOD = 0;

        for (int i = 0; i < thresholds.size(); i++) {
            float threshold = thresholds[i];

            // 현재 LOD보다 낮은 LOD로 전환 시: 더 멀리 가야 함
            if (i >= currentLOD) {
                threshold *= (1.0f + hysteresis);
            }
            // 현재 LOD보다 높은 LOD로 전환 시: 더 가까이 와야 함
            else {
                threshold *= (1.0f - hysteresis);
            }

            if (distance <= threshold) {
                newLOD = i;
                break;
            }
            newLOD = i + 1;
        }

        currentLOD = newLOD;
        return currentLOD;
    }
};

해결책 2: Alpha 블렌딩 전환

부드러운 LOD 전환
class SmoothLODTransition {
private:
    float transitionProgress = 0.0f;
    int fromLOD = 0;
    int toLOD = 0;
    bool transitioning = false;
    float transitionSpeed = 2.0f;  // 초당 전환 속도

public:
    void update(float deltaTime, int targetLOD) {
        if (targetLOD != toLOD && !transitioning) {
            // 새로운 전환 시작
            fromLOD = toLOD;
            toLOD = targetLOD;
            transitioning = true;
            transitionProgress = 0.0f;
        }

        if (transitioning) {
            transitionProgress += deltaTime * transitionSpeed;
            if (transitionProgress >= 1.0f) {
                transitionProgress = 1.0f;
                transitioning = false;
                fromLOD = toLOD;
            }
        }
    }

    void render(const Camera& camera) {
        if (transitioning) {
            // 두 LOD를 블렌딩하여 렌더링
            float alpha = smoothstep(0.0f, 1.0f, transitionProgress);

            renderMesh(getLODMesh(fromLOD), 1.0f - alpha);
            renderMesh(getLODMesh(toLOD), alpha);
        } else {
            renderMesh(getLODMesh(toLOD), 1.0f);
        }
    }

    float smoothstep(float edge0, float edge1, float x) {
        x = std::clamp((x - edge0) / (edge1 - edge0), 0.0f, 1.0f);
        return x * x * (3 - 2 * x);
    }
};

Part 5: 연속 LOD (CLOD)

미리 만든 LOD 메시 외에 다른 방법은?

연속 LOD(CLOD)는 런타임에 메시 해상도를 동적으로 조절합니다.

Progressive Mesh 기반 CLOD

    int vertexFrom;    // 제거될 정점

    void precompute() {

        while (triangles.size() > minTriangles) {
            EdgeCollapse bestCollapse = findLowestErrorCollapse();
            collapseSequence.push_back(bestCollapse);
            applyCollapse(bestCollapse);
        }

            (1.0f - t) * collapseSequence.size());

            applyCollapse(collapseSequence[currentDetail]);

Part 6: HLOD와 Impostor

도시 같은 복잡한 씬은 어떻게 처리하나요?

HLOD(Hierarchical LOD)는 여러 물체를 하나로 병합합니다.

가까움 (개별 렌더링)
건물1, 건물2, 건물3, 건물4
4 Draw Calls
→
멀리 (HLOD 병합)
건물 그룹 (병합)
1 Draw Call

Impostor (빌보드):

Impostor 시스템
class ImpostorSystem {
private:
    struct Impostor {
        int objectId;
        std::vector<Texture> views;  // 여러 각도에서 캡처한 이미지
        float captureDistance;
    };

    std::unordered_map<int, Impostor> impostors;

public:
    // 오프라인: 여러 각도에서 물체 렌더링하여 텍스처 생성
    void generateImpostor(Object& object, int viewCount = 8) {
        Impostor impostor;
        impostor.objectId = object.id;

        for (int i = 0; i < viewCount; i++) {
            float angle = (2.0f * M_PI * i) / viewCount;

            // 해당 각도에서 렌더링
            Texture capture = renderToTexture(object, angle);
            impostor.views.push_back(capture);
        }

        impostors[object.id] = impostor;
    }

    // 런타임: 카메라 각도에 맞는 빌보드 렌더링
    void renderImpostor(int objectId, const Vec3& objectPos,
                        const Camera& camera) {
        Impostor& impostor = impostors[objectId];

        // 카메라에서 물체를 보는 각도 계산
        Vec3 toObject = objectPos - camera.position;
        float angle = atan2(toObject.x, toObject.z);

        // 가장 가까운 뷰 선택 (또는 두 뷰 블렌딩)
        int viewIndex = selectBestView(angle, impostor.views.size());

        // 빌보드 렌더링
        renderBillboard(objectPos, impostor.views[viewIndex], camera);
    }
};

📌 핵심 정리

  • LOD: 거리/화면 점유율에 따라 메시 디테일 조절
  • 팝핑 방지: 히스테리시스, 알파 블렌딩으로 부드러운 전환
  • CLOD: Progressive Mesh로 연속적 디테일 조절
  • HLOD: 여러 물체를 그룹으로 병합하여 드로우 콜 감소
  • Impostor: 매우 먼 물체는 빌보드 텍스처로 대체
