Light Culling 기법 (Tiled, Clustered)

Part 1: 다수의 라이트 문제

라이트가 수천 개면 어떤 문제가 있나요?

기본 Deferred에서 각 라이트마다 전체 화면을 처리하면 O(픽셀 × 라이트)가 됩니다.

라이트 1000개, 1080p = 20억 번의 라이팅 계산!

Light Culling은 각 픽셀에 영향을 주는 라이트만 계산합니다.

화면 (모든 픽셀)
모든 픽셀이 모든 라이트를 체크해야 함
×
라이트들 (1000개)
대부분은 픽셀에 영향 안 줌 (낭비)
=
O(픽셀 x 라이트)
1080p + 1000개 = 20억 번 계산!

Part 2: Tiled Deferred Rendering

Tiled Deferred는 어떻게 동작하나요?

화면을 타일(예: 16×16 픽셀)로 나누고, 각 타일에 영향을 주는 라이트 목록을 미리 계산합니다.

1. 화면을 타일로 분할 (16x16)
T T T T T T
T T T T T T
T T T T T T
↓
2. 타일별 라이트 컬링
타일 A: 3개만 영향
타일 B: 1개만 영향
타일 C: 5개 영향
↓
3. 라이팅 패스
각 타일은 자신의 라이트 리스트만 처리
O(픽셀 x 타일당 라이트) << O(픽셀 x 전체 라이트)

Tiled 접근 방식

타일 Light Culling (Compute Shader)
// 타일 크기
#define TILE_SIZE 16
#define MAX_LIGHTS_PER_TILE 256

// 타일별 라이트 목록
groupshared uint tileLightCount;
groupshared uint tileLightIndices[MAX_LIGHTS_PER_TILE];

// 타일의 깊이 범위 (절두체 좁히기)
groupshared uint tileMinZ;
groupshared uint tileMaxZ;

[numthreads(TILE_SIZE, TILE_SIZE, 1)]
void CS_LightCulling(uint3 groupID : SV_GroupID,
                     uint3 groupThreadID : SV_GroupThreadID,
                     uint groupIndex : SV_GroupIndex)
{
    // 1. 초기화 (첫 스레드만)
    if (groupIndex == 0)
    {
        tileLightCount = 0;
        tileMinZ = 0xFFFFFFFF;
        tileMaxZ = 0;
    }
    GroupMemoryBarrierWithGroupSync();

    // 2. 타일 내 깊이 범위 계산
    uint2 pixelCoord = groupID.xy * TILE_SIZE + groupThreadID.xy;
    float depth = depthBuffer[pixelCoord];
    float linearDepth = LinearizeDepth(depth);

    // Atomic으로 Min/Max 업데이트
    uint depthUint = asuint(linearDepth);
    InterlockedMin(tileMinZ, depthUint);
    InterlockedMax(tileMaxZ, depthUint);

    GroupMemoryBarrierWithGroupSync();

    // 3. 타일 절두체 구성
    float minZ = asfloat(tileMinZ);
    float maxZ = asfloat(tileMaxZ);

    // 타일 경계 → 뷰 공간 절두체
    Frustum tileFrustum = CreateTileFrustum(
        groupID.xy, TILE_SIZE, minZ, maxZ);

    // 4. 라이트 컬링 (스레드들이 분담)
    uint lightCount = totalLightCount;
    for (uint i = groupIndex; i < lightCount; i += TILE_SIZE * TILE_SIZE)
    {
        Light light = lights[i];

        // 라이트가 타일 절두체와 교차하는지
        if (IntersectFrustumSphere(tileFrustum, light.position, light.radius))
        {
            uint index;
            InterlockedAdd(tileLightCount, 1, index);

            if (index < MAX_LIGHTS_PER_TILE)
            {
                tileLightIndices[index] = i;
            }
        }
    }

    GroupMemoryBarrierWithGroupSync();

    // 5. 라이트 리스트 저장
    if (groupIndex == 0)
    {
        uint tileIndex = groupID.y * numTilesX + groupID.x;
        tileInfo[tileIndex].lightCount = min(tileLightCount, MAX_LIGHTS_PER_TILE);
        tileInfo[tileIndex].lightOffset = tileIndex * MAX_LIGHTS_PER_TILE;
    }

    // 모든 스레드가 라이트 인덱스 저장
    if (groupIndex < tileLightCount && groupIndex < MAX_LIGHTS_PER_TILE)
    {
        uint tileIndex = groupID.y * numTilesX + groupID.x;
        lightIndexList[tileIndex * MAX_LIGHTS_PER_TILE + groupIndex] =
            tileLightIndices[groupIndex];
    }
}

Part 3: Tiled 라이팅 패스

컬링 후 라이팅은 어떻게 하나요?

각 픽셀은 자신이 속한 타일의 라이트 리스트만 순회합니다.

Tiled Lighting (Compute Shader)
[numthreads(TILE_SIZE, TILE_SIZE, 1)]
void CS_TiledLighting(uint3 groupID : SV_GroupID,
                      uint3 dispatchThreadID : SV_DispatchThreadID)
{
    uint2 pixelCoord = dispatchThreadID.xy;
    uint tileIndex = groupID.y * numTilesX + groupID.x;

    // G-Buffer 읽기
    float3 albedo = gBuffer0[pixelCoord].rgb;
    float3 normal = DecodeNormal(gBuffer1[pixelCoord].rg);
    float3 worldPos = ReconstructPosition(pixelCoord);
    float roughness = gBuffer1[pixelCoord].b;
    float metallic = gBuffer1[pixelCoord].a;

    // 타일의 라이트 정보
    uint lightCount = tileInfo[tileIndex].lightCount;
    uint lightOffset = tileInfo[tileIndex].lightOffset;

    // 라이팅 계산
    float3 finalColor = float3(0, 0, 0);
    float3 viewDir = normalize(cameraPos - worldPos);

    // 해당 타일의 라이트만 순회!
    for (uint i = 0; i < lightCount; i++)
    {
        uint lightIndex = lightIndexList[lightOffset + i];
        Light light = lights[lightIndex];

        float3 lightDir = light.position - worldPos;
        float distance = length(lightDir);
        lightDir /= distance;

        // 거리 감쇠
        float attenuation = 1.0 / (distance * distance);
        attenuation *= saturate(1.0 - distance / light.radius);

        // PBR 라이팅
        finalColor += CalculatePBR(albedo, normal, viewDir, lightDir,
                                    roughness, metallic, light.color)
                      * attenuation;
    }

    outputTexture[pixelCoord] = float4(finalColor, 1);
}

Part 4: Clustered Deferred

Tiled의 한계는 무엇인가요?

Tiled는 2D 타일이라 깊이 방향 라이트 분포를 잘 처리하지 못합니다.

Clustered는 깊이 방향으로도 분할해 3D 클러스터를 만듭니다.

Tiled (2D 분할)
타일 = 2D, 깊이 무시
라이트가 화면 전체에 영향
→ 전체 타일에 포함됨
vs
Clustered (3D 분할)
X, Y + 깊이(Z) 슬라이스
Z0 / Z1 / Z2 별 분할
→ 정확한 3D 컬링

Tiled vs Clustered 비교

Clustered 데이터 구조
// 클러스터 구조
// X, Y: 화면 타일 (예: 16×16)
// Z: 깊이 슬라이스 (예: 24개, 로그 스케일)

struct ClusterInfo
{
    uint lightCount;
    uint lightOffset;
};

// 깊이 슬라이스 계산 (로그 분포)
uint GetDepthSlice(float linearDepth)
{
    // 로그 분포: 가까울수록 촘촘, 멀수록 성김
    float logDepth = log(linearDepth / nearPlane);
    float logRange = log(farPlane / nearPlane);
    return uint(logDepth / logRange * numDepthSlices);
}

// 클러스터 인덱스 계산
uint3 GetClusterIndex(uint2 pixelCoord, float linearDepth)
{
    uint3 cluster;
    cluster.x = pixelCoord.x / TILE_SIZE;
    cluster.y = pixelCoord.y / TILE_SIZE;
    cluster.z = GetDepthSlice(linearDepth);
    return cluster;
}

uint GetClusterLinearIndex(uint3 cluster)
{
    return cluster.x +
           cluster.y * numClustersX +
           cluster.z * numClustersX * numClustersY;
}

Part 5: Clustered 컬링

3D 클러스터 컬링은 어떻게 하나요?

각 클러스터의 AABB를 구하고, 라이트 볼륨과 교차 테스트합니다.

Clustered Light Culling
// 클러스터 AABB 계산
AABB GetClusterAABB(uint3 clusterIndex)
{
    float2 tileMin = float2(clusterIndex.xy) * TILE_SIZE;
    float2 tileMax = tileMin + TILE_SIZE;

    // 스크린 → NDC
    float2 ndcMin = tileMin / screenSize * 2.0 - 1.0;
    float2 ndcMax = tileMax / screenSize * 2.0 - 1.0;

    // 깊이 범위 (로그 분포)
    float zNear = GetDepthAtSlice(clusterIndex.z);
    float zFar = GetDepthAtSlice(clusterIndex.z + 1);

    // 뷰 공간 AABB 계산
    AABB aabb;
    aabb.minPoint = ScreenToView(ndcMin, zNear);
    aabb.maxPoint = ScreenToView(ndcMax, zFar);

    return aabb;
}

[numthreads(CLUSTERS_X, CLUSTERS_Y, CLUSTERS_Z)]
void CS_ClusteredCulling(uint3 clusterID : SV_DispatchThreadID)
{
    uint clusterIndex = GetClusterLinearIndex(clusterID);

    // 클러스터 AABB
    AABB clusterAABB = GetClusterAABB(clusterID);

    // 라이트 컬링
    uint lightCount = 0;
    uint lightIndices[MAX_LIGHTS_PER_CLUSTER];

    for (uint i = 0; i < totalLightCount; i++)
    {
        Light light = lights[i];

        // AABB vs Sphere 교차 테스트
        if (IntersectAABBSphere(clusterAABB, light.position, light.radius))
        {
            if (lightCount < MAX_LIGHTS_PER_CLUSTER)
            {
                lightIndices[lightCount++] = i;
            }
        }
    }

    // 결과 저장
    clusterInfo[clusterIndex].lightCount = lightCount;
    // ... 라이트 인덱스 저장
}
기법 장점 단점
Tiled Deferred 구현 간단, 메모리 적음 깊이 변화 큰 씬에서 비효율
Clustered Deferred 3D 컬링으로 정확함 메모리 더 사용, 구현 복잡
Clustered Forward 반투명 지원, MSAA 쉬움 셰이더에서 컬링 데이터 접근

Part 6: Forward+ (Tiled Forward)

Forward 렌더링에서도 Light Culling이 가능한가요?

Forward+는 Tiled/Clustered 컬링을 Forward 렌더링과 결합합니다. Deferred의 한계(반투명, MSAA)를 극복합니다.

1. Depth Pre-Pass
깊이만 먼저 렌더링
↓
2. Light Culling (Compute)
Tiled/Clustered 컬링
타일별 라이트 리스트 생성
↓
3. Forward Shading
각 픽셀에서 타일 라이트 리스트 접근
해당 라이트만 계산
반투명, MSAA 지원!

Forward의 유연성 + Deferred의 라이트 효율

Forward+ 파이프라인

Forward+ Pixel Shader
// Forward+ 라이팅 (Pixel Shader)
float4 PS_ForwardPlus(VSOutput input) : SV_Target
{
    // 픽셀이 속한 타일/클러스터 계산
    uint2 tileCoord = uint2(input.position.xy) / TILE_SIZE;
    uint tileIndex = tileCoord.y * numTilesX + tileCoord.x;

    // 머티리얼 샘플링
    float3 albedo = albedoMap.Sample(samp, input.uv).rgb;
    float3 normal = normalMap.Sample(samp, input.uv).rgb * 2.0 - 1.0;
    // ...

    // 라이팅 계산
    float3 finalColor = ambientLight * albedo;
    float3 viewDir = normalize(cameraPos - input.worldPos);

    // 타일의 라이트만 순회
    uint lightCount = tileLightInfo[tileIndex].lightCount;
    uint lightOffset = tileLightInfo[tileIndex].lightOffset;

    for (uint i = 0; i < lightCount; i++)
    {
        uint lightIndex = lightIndexList[lightOffset + i];
        Light light = lights[lightIndex];

        // PBR 라이팅
        finalColor += CalculatePBR(/* ... */);
    }

    return float4(finalColor, 1);
}
🎯 렌더링 기법 선택 가이드

Tiled Deferred: 다수의 라이트, 불투명 위주

Clustered Deferred: 복잡한 깊이 분포, 대규모 씬

Forward+: 반투명 많음, MSAA 필요

Clustered Forward: 현대 엔진의 표준 (언리얼, 유니티)

📌 핵심 정리

  • Light Culling: 픽셀에 영향 주는 라이트만 계산해 성능 향상
  • Tiled: 화면을 2D 타일로 분할, 타일별 라이트 리스트
  • Clustered: 3D 클러스터로 깊이 방향도 분할, 더 정확한 컬링
  • Forward+: Light Culling + Forward = 반투명/MSAA 지원
  • 현대 엔진은 Clustered Forward/Deferred가 표준
