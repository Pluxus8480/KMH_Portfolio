G-Buffer 구조와 최적화

Part 1: G-Buffer 복습

G-Buffer란 무엇인가요?

G-Buffer (Geometry Buffer)는 Deferred Rendering에서 라이팅에 필요한 기하 정보를 저장하는 버퍼 집합입니다.

첫 번째 패스에서 정보를 저장하고, 두 번째 패스에서 라이팅을 계산합니다.

Part 2: 일반적인 G-Buffer 레이아웃

G-Buffer에는 어떤 정보가 필요한가요?

PBR 라이팅에 필요한 정보: 위치(깊이), 법선, 알베도, 러프니스, 메탈릭, 이미시브 등

전형적인 G-Buffer 구성 (~256 bits/pixel)
RT0 (32 bits)
R: Albedo (8)G: Albedo (8)B: Albedo (8)A: Metallic (8)
RT1 (64 bits)
R: Normal.x (16)G: Normal.y (16)B: Normal.z (16)A: Roughness (16)
RT2 (64 bits)
R: Emissive.r (16)G: Emissive.g (16)B: Emissive.b (16)A: AO (16)
Depth (32 bits)
Depth 24-32 bits
R8G8B8A8_UNORM | R16G16B16A16_FLOAT | D24S8 or D32F

Part 3: 대역폭 최적화

G-Buffer가 너무 크면 어떤 문제가 있나요?

G-Buffer 크기 = 메모리 대역폭입니다. 1080p에서 4개 RT는 프레임당 ~30MB를 읽고 씁니다.

Part 4: 법선 인코딩

법선을 어떻게 압축하나요?

법선은 단위 벡터이므로 x, y만 저장하고 z를 재구성할 수 있습니다. 여러 인코딩 방식이 있습니다.

법선 인코딩 방식들
// 1. 단순 XY 저장 (Z 재구성)
// 문제: 반구만 표현 가능
float2 EncodeNormalXY(float3 n)
{
    return n.xy;
}

float3 DecodeNormalXY(float2 enc)
{
    float3 n;
    n.xy = enc;
    n.z = sqrt(1.0 - saturate(dot(n.xy, n.xy)));
    return n;
}

// 2. Spheremap Transform (더 정확)
float2 EncodeNormalSpheremap(float3 n)
{
    float f = sqrt(8.0 * n.z + 8.0);
    return n.xy / f + 0.5;
}

float3 DecodeNormalSpheremap(float2 enc)
{
    float2 fenc = enc * 4.0 - 2.0;
    float f = dot(fenc, fenc);
    float g = sqrt(1.0 - f / 4.0);
    float3 n;
    n.xy = fenc * g;
    n.z = 1.0 - f / 2.0;
    return n;
}

// 3. Octahedron Encoding (현재 가장 인기)
float2 OctWrap(float2 v)
{
    return (1.0 - abs(v.yx)) * (v.xy >= 0.0 ? 1.0 : -1.0);
}

float2 EncodeNormalOctahedron(float3 n)
{
    n /= (abs(n.x) + abs(n.y) + abs(n.z));
    n.xy = n.z >= 0.0 ? n.xy : OctWrap(n.xy);
    n.xy = n.xy * 0.5 + 0.5;
    return n.xy;
}

float3 DecodeNormalOctahedron(float2 enc)
{
    enc = enc * 2.0 - 1.0;
    float3 n = float3(enc.xy, 1.0 - abs(enc.x) - abs(enc.y));
    float t = saturate(-n.z);
    n.xy += n.xy >= 0.0 ? -t : t;
    return normalize(n);
}
인코딩 저장 크기 정확도 연산 비용
RGB 저장 48 bits 완벽 없음
XY + Z재구성 32 bits 낮음 (반구만) 낮음
Spheremap 32 bits 중간 중간
Octahedron 32 bits 높음 낮음

Part 5: 최적화된 G-Buffer 레이아웃

실전에서 사용하는 최적화된 레이아웃은?

2~3개의 렌더 타겟으로 압축하는 것이 일반적입니다:

최적화된 G-Buffer (80~112 bits/pixel)
RT0 (32 bits)
R: Albedo sRGB (8)G: Albedo sRGB (8)B: Albedo sRGB (8)A: Specular (8)
RT1 (32 bits)
R: Normal.x Octahedron (16)G: Normal.y Octahedron (16)
RT2 (16 bits)
R: Metallic (8)G: Roughness (8)
Depth (32 bits)
D24S8 (32 bits)
대역폭 절감
192 bits → 112 bits (42% 감소!)

최적화된 G-Buffer

최적화된 G-Buffer 셰이더
// Geometry Pass - G-Buffer 출력
struct GBufferOutput
{
    float4 RT0 : SV_Target0;  // Albedo + Specular
    float2 RT1 : SV_Target1;  // Normal (Octahedron)
    float2 RT2 : SV_Target2;  // Metallic + Roughness
};

GBufferOutput PS_GBuffer(VSOutput input)
{
    GBufferOutput output;

    // 알베도
    float3 albedo = albedoTexture.Sample(samp, input.uv).rgb;

    // 법선 (노말맵 적용 후)
    float3 normal = normalTexture.Sample(samp, input.uv).rgb * 2.0 - 1.0;
    normal = mul(normal, input.TBN);
    normal = normalize(normal);

    // 머티리얼
    float metallic = metallicTexture.Sample(samp, input.uv).r;
    float roughness = roughnessTexture.Sample(samp, input.uv).r;

    // 패킹
    output.RT0 = float4(albedo, 0);  // A 채널 활용 가능
    output.RT1 = EncodeNormalOctahedron(normal);
    output.RT2 = float2(metallic, roughness);

    return output;
}

// Lighting Pass - G-Buffer 언패킹
struct GBufferData
{
    float3 albedo;
    float3 normal;
    float metallic;
    float roughness;
    float3 worldPos;
};

GBufferData UnpackGBuffer(float2 uv, float2 screenPos)
{
    GBufferData data;

    // 읽기
    float4 rt0 = gBuffer0.Sample(pointSamp, uv);
    float2 rt1 = gBuffer1.Sample(pointSamp, uv);
    float2 rt2 = gBuffer2.Sample(pointSamp, uv);
    float depth = depthBuffer.Sample(pointSamp, uv).r;

    // 언패킹
    data.albedo = rt0.rgb;
    data.normal = DecodeNormalOctahedron(rt1);
    data.metallic = rt2.r;
    data.roughness = rt2.g;

    // 깊이에서 월드 위치 재구성
    data.worldPos = ReconstructWorldPosition(screenPos, depth);

    return data;
}

Part 6: 위치 재구성

World Position을 저장하지 않으려면?

깊이 버퍼와 역 뷰-프로젝션 행렬로 월드 위치를 재구성합니다. 3채널 대신 1채널로 충분합니다.

깊이에서 위치 재구성
// 방법 1: 역행렬 사용
float3 ReconstructWorldPosition(float2 screenUV, float depth)
{
    // 스크린 UV → NDC (-1 ~ 1)
    float2 ndc = screenUV * 2.0 - 1.0;
    ndc.y = -ndc.y;  // DirectX Y 반전

    // 클립 공간 위치
    float4 clipPos = float4(ndc, depth, 1.0);

    // 역 뷰-프로젝션 변환
    float4 worldPos = mul(clipPos, invViewProjMatrix);
    worldPos /= worldPos.w;  // 원근 나눗셈

    return worldPos.xyz;
}

// 방법 2: 레이 + 선형 깊이 (더 빠름)
float3 ReconstructWorldPositionFast(float2 screenUV, float depth)
{
    // 카메라에서 픽셀로의 방향 벡터 (미리 계산)
    float3 viewRay = GetViewRay(screenUV);

    // 선형 깊이로 변환
    float linearDepth = LinearizeDepth(depth, nearPlane, farPlane);

    // 카메라 위치 + 방향 × 거리
    return cameraPosition + viewRay * linearDepth;
}

// 뷰 레이 계산 (Vertex Shader에서)
float3 GetViewRay(float2 screenUV)
{
    // 카메라 코너에서 보간
    // 전체 화면 쿼드의 각 정점에 뷰 레이 저장
    float3 topLeft = frustumCorners[0];
    float3 topRight = frustumCorners[1];
    float3 bottomLeft = frustumCorners[2];
    float3 bottomRight = frustumCorners[3];

    float3 top = lerp(topLeft, topRight, screenUV.x);
    float3 bottom = lerp(bottomLeft, bottomRight, screenUV.x);
    return lerp(top, bottom, screenUV.y);
}

📌 핵심 정리

  • G-Buffer는 Deferred Rendering의 핵심, 대역폭이 성능 병목
  • Octahedron 인코딩으로 법선을 2채널로 압축
  • 깊이 재구성으로 World Position 저장 불필요
  • 데이터 패킹으로 렌더 타겟 수와 크기 최소화
  • 최적화로 192 bits → 80~112 bits까지 압축 가능
